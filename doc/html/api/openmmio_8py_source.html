00030 def get_dipole(simulation,q=None,positions=None):
00031     """Return the current dipole moment in Debye.
00032     Note that this quantity is meaningless if the system carries a net charge."""
00033     dx = 0.0
00034     dy = 0.0
00035     dz = 0.0
00036     enm_debye = 48.03204255928332 # Conversion factor from e*nm to Debye
00037     for i in simulation.system.getForces():
00038         if i.__class__.__name__ == "AmoebaMultipoleForce":
00039             mm = i.getSystemMultipoleMoments(simulation.context)
00040             dx += mm[1]
00041             dy += mm[2]
00042             dz += mm[3]
00043         if i.__class__.__name__ == "NonbondedForce":
00044             # Get array of charges.
00045             if q == None:
00046                 q = np.array([i.getParticleParameters(j)[0]._value for j in range(i.getNumParticles())])
00047             # Get array of positions in nanometers.
00048             if positions == None:
00049                 positions = simulation.context.getState(getPositions=True).getPositions()
00050             #x = np.array([j._value for j in positions])
00051             x = np.array(positions.value_in_unit(nanometer))
00052             # Multiply charges by positions to get dipole moment.
00053             dip = enm_debye * np.sum(x*q.reshape(-1,1),axis=0)
00054             dx += dip[0]
00055             dy += dip[1]
00056             dz += dip[2]
00057     return [dx,dy,dz]

00182 def MTSVVVRIntegrator(temperature, collision_rate, timestep, system, ninnersteps=4):
00183     """
00184     Create a multiple timestep velocity verlet with velocity randomization (VVVR) integrator.
00185     
00186     ARGUMENTS
00187 
00188     temperature (numpy.unit.Quantity compatible with kelvin) - the temperature
00189     collision_rate (numpy.unit.Quantity compatible with 1/picoseconds) - the collision rate
00190     timestep (numpy.unit.Quantity compatible with femtoseconds) - the integration timestep
00191     system (simtk.openmm.System) - system whose forces will be partitioned
00192     ninnersteps (int) - number of inner timesteps (default: 4)
00193 
00194     RETURNS
00195 
00196     integrator (openmm.CustomIntegrator) - a VVVR integrator
00197 
00198     NOTES
00199     
00200     This integrator is equivalent to a Langevin integrator in the velocity Verlet discretization with a
00201     timestep correction to ensure that the field-free diffusion constant is timestep invariant.  The inner
00202     velocity Verlet discretization is transformed into a multiple timestep algorithm.
00203 
00204     REFERENCES
00205 
00206     VVVR Langevin integrator: 
00207     * http://arxiv.org/abs/1301.3800
00208     * http://arxiv.org/abs/1107.2967 (to appear in PRX 2013)    
00209     
00210     TODO
00211 
00212     Move initialization of 'sigma' to setting the per-particle variables.
00213     
00214     """
00215     # Multiple timestep Langevin integrator.
00216     for i in system.getForces():
00217         if i.__class__.__name__ in ["NonbondedForce", "CustomNonbondedForce", "AmoebaVdwForce", "AmoebaMultipoleForce"]:
00218             # Slow force.
00219             print i.__class__.__name__, "is a Slow Force"
00220             i.setForceGroup(1)
00221         else:
00222             print i.__class__.__name__, "is a Fast Force"
00223             # Fast force.
00224             i.setForceGroup(0)
00225 
00226     kB = BOLTZMANN_CONSTANT_kB * AVOGADRO_CONSTANT_NA
00227     kT = kB * temperature
00228     
00229     integrator = openmm.CustomIntegrator(timestep)
00230     
00231     integrator.addGlobalVariable("dt_fast", timestep/float(ninnersteps)) # fast inner timestep
00232     integrator.addGlobalVariable("kT", kT) # thermal energy
00233     integrator.addGlobalVariable("a", numpy.exp(-collision_rate*timestep)) # velocity mixing parameter
00234     integrator.addGlobalVariable("b", numpy.sqrt((2/(collision_rate*timestep)) * numpy.tanh(collision_rate*timestep/2))) # timestep correction parameter
00235     integrator.addPerDofVariable("sigma", 0) 
00236     integrator.addPerDofVariable("x1", 0) # position before application of constraints
00237 
00238     #
00239     # Pre-computation.
00240     # This only needs to be done once, but it needs to be done for each degree of freedom.
00241     # Could move this to initialization?
00242     #
00243     integrator.addComputePerDof("sigma", "sqrt(kT/m)")
00244 
00245     # 
00246     # Velocity perturbation.
00247     #
00248     integrator.addComputePerDof("v", "sqrt(a)*v + sqrt(1-a)*sigma*gaussian")
00249     integrator.addConstrainVelocities();
00250     
00251     #
00252     # Symplectic inner multiple timestep.
00253     #
00254     integrator.addUpdateContextState(); 
00255     integrator.addComputePerDof("v", "v + 0.5*b*dt*f1/m")
00256     for innerstep in range(ninnersteps):
00257         # Fast inner symplectic timestep.
00258         integrator.addComputePerDof("v", "v + 0.5*b*dt_fast*f0/m")
00259         integrator.addComputePerDof("x", "x + v*b*dt_fast")
00260         integrator.addComputePerDof("x1", "x")
00261         integrator.addConstrainPositions();        
00262         integrator.addComputePerDof("v", "v + 0.5*b*dt_fast*f0/m + (x-x1)/dt_fast")
00263     integrator.addComputePerDof("v", "v + 0.5*b*dt*f1/m") # TODO: Additional velocity constraint correction?
00264     integrator.addConstrainVelocities();
00265 
00266     #
00267     # Velocity randomization
00268     #
00269     integrator.addComputePerDof("v", "sqrt(a)*v + sqrt(1-a)*sigma*gaussian")
00270     integrator.addConstrainVelocities();
00271 
00272     return integrator
