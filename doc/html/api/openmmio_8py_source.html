45 def get_multipoles(simulation,q=None,mass=None,positions=None,rmcom=True):
46  """Return the current multipole moments in Debye and Buckingham units. """
47  dx = 0.0
48  dy = 0.0
49  dz = 0.0
50  qxx = 0.0
51  qxy = 0.0
52  qxz = 0.0
53  qyy = 0.0
54  qyz = 0.0
55  qzz = 0.0
56  enm_debye = 48.03204255928332 # Conversion factor from e*nm to Debye
57  for i in simulation.system.getForces():
58  if isinstance(i, AmoebaMultipoleForce):
59  mm = i.getSystemMultipoleMoments(simulation.context)
60  dx += mm[1]
61  dy += mm[2]
62  dz += mm[3]
63  qxx += mm[4]
64  qxy += mm[5]
65  qxz += mm[6]
66  qyy += mm[8]
67  qyz += mm[9]
68  qzz += mm[12]
69  if isinstance(i, NonbondedForce):
70  # Get array of charges.
71  if q == None:
72  q = np.array([i.getParticleParameters(j)[0]._value for j in range(i.getNumParticles())])
73  # Get array of positions in nanometers.
74  if positions == None:
75  positions = simulation.context.getState(getPositions=True).getPositions()
76  if mass == None:
77  mass = np.array([simulation.context.getSystem().getParticleMass(k).value_in_unit(dalton) \
78  for k in range(simulation.context.getSystem().getNumParticles())])
79  x = np.array(positions.value_in_unit(nanometer))
80  if rmcom:
81  com = np.sum(x*mass.reshape(-1,1),axis=0) / np.sum(mass)
82  x -= com
83  xx, xy, xz, yy, yz, zz = (x[:,k]*x[:,l] for k, l in [(0,0),(0,1),(0,2),(1,1),(1,2),(2,2)])
84  # Multiply charges by positions to get dipole moment.
85  dip = enm_debye * np.sum(x*q.reshape(-1,1),axis=0)
86  dx += dip[0]
87  dy += dip[1]
88  dz += dip[2]
89  qxx += enm_debye * 15 * np.sum(q*xx)
90  qxy += enm_debye * 15 * np.sum(q*xy)
91  qxz += enm_debye * 15 * np.sum(q*xz)
92  qyy += enm_debye * 15 * np.sum(q*yy)
93  qyz += enm_debye * 15 * np.sum(q*yz)
94  qzz += enm_debye * 15 * np.sum(q*zz)
95  tr = qxx+qyy+qzz
96  qxx -= tr/3
97  qyy -= tr/3
98  qzz -= tr/3
99  # This ordering has to do with the way TINKER prints it out.
100  return [dx,dy,dz,qxx,qxy,qyy,qxz,qyz,qzz]

316 def MTSVVVRIntegrator(temperature, collision_rate, timestep, system, ninnersteps=4):
317  """
318  Create a multiple timestep velocity verlet with velocity randomization (VVVR) integrator.
319 
320  ARGUMENTS
321 
322  temperature (Quantity compatible with kelvin) - the temperature
323  collision_rate (Quantity compatible with 1/picoseconds) - the collision rate
324  timestep (Quantity compatible with femtoseconds) - the integration timestep
325  system (simtk.openmm.System) - system whose forces will be partitioned
326  ninnersteps (int) - number of inner timesteps (default: 4)
327 
328  RETURNS
329 
330  integrator (openmm.CustomIntegrator) - a VVVR integrator
331 
332  NOTES
333 
334  This integrator is equivalent to a Langevin integrator in the velocity Verlet discretization with a
335  timestep correction to ensure that the field-free diffusion constant is timestep invariant. The inner
336  velocity Verlet discretization is transformed into a multiple timestep algorithm.
337 
338  REFERENCES
339 
340  VVVR Langevin integrator:
341  * http://arxiv.org/abs/1301.3800
342  * http://arxiv.org/abs/1107.2967 (to appear in PRX 2013)
343 
344  TODO
345 
346  Move initialization of 'sigma' to setting the per-particle variables.
347 
348  """
349  # Multiple timestep Langevin integrator.
350  for i in system.getForces():
351  if i.__class__.__name__ in ["NonbondedForce", "CustomNonbondedForce", "AmoebaVdwForce", "AmoebaMultipoleForce"]:
352  # Slow force.
353  logger.info(i.__class__.__name__ + " is a Slow Force\n")
354  i.setForceGroup(1)
355  else:
356  logger.info(i.__class__.__name__ + " is a Fast Force\n")
357  # Fast force.
358  i.setForceGroup(0)
359 
360  kB = BOLTZMANN_CONSTANT_kB * AVOGADRO_CONSTANT_NA
361  kT = kB * temperature
362 
363  integrator = CustomIntegrator(timestep)
364 
365  integrator.addGlobalVariable("dt_fast", timestep/float(ninnersteps)) # fast inner timestep
366  integrator.addGlobalVariable("kT", kT) # thermal energy
367  integrator.addGlobalVariable("a", np.exp(-collision_rate*timestep)) # velocity mixing parameter
368  integrator.addGlobalVariable("b", np.sqrt((2/(collision_rate*timestep)) * np.tanh(collision_rate*timestep/2))) # timestep correction parameter
369  integrator.addPerDofVariable("sigma", 0)
370  integrator.addPerDofVariable("x1", 0) # position before application of constraints
371 
372  #
373  # Pre-computation.
374  # This only needs to be done once, but it needs to be done for each degree of freedom.
375  # Could move this to initialization?
376  #
377  integrator.addComputePerDof("sigma", "sqrt(kT/m)")
378 
379  #
380  # Velocity perturbation.
381  #
382  integrator.addComputePerDof("v", "sqrt(a)*v + sqrt(1-a)*sigma*gaussian")
383  integrator.addConstrainVelocities();
384 
385  #
386  # Symplectic inner multiple timestep.
387  #
388  integrator.addUpdateContextState();
389  integrator.addComputePerDof("v", "v + 0.5*b*dt*f1/m")
390  for innerstep in range(ninnersteps):
391  # Fast inner symplectic timestep.
392  integrator.addComputePerDof("v", "v + 0.5*b*dt_fast*f0/m")
393  integrator.addComputePerDof("x", "x + v*b*dt_fast")
394  integrator.addComputePerDof("x1", "x")
395  integrator.addConstrainPositions();
396  integrator.addComputePerDof("v", "v + 0.5*b*dt_fast*f0/m + (x-x1)/dt_fast")
397  integrator.addComputePerDof("v", "v + 0.5*b*dt*f1/m") # TODO: Additional velocity constraint correction?
398  integrator.addConstrainVelocities();
399 
400  #
401  # Velocity randomization
402  #
403  integrator.addComputePerDof("v", "sqrt(a)*v + sqrt(1-a)*sigma*gaussian")
404  integrator.addConstrainVelocities();
405 
406  return integrator

832  def optimize(self, shot=0, crit=1e-4):
833 
834  """ Optimize the geometry and align the optimized geometry to the starting geometry, and return the RMSD. """
835 
836  steps = int(max(1, -1*np.log10(crit)))
837  self.update_simulation()
838  self.set_positions(shot)
839  # Get the previous geometry.
840  X0 = np.array([j for i, j in enumerate(self.simulation.context.getState(getPositions=True).getPositions().value_in_unit(angstrom)) if self.AtomMask[i]])
841  # Minimize the energy. Optimizer works best in "steps".
842  for logc in np.linspace(0, np.log10(crit), steps):
843  self.simulation.minimizeEnergy(tolerance=10**logc*kilojoule/mole)
844  # Get the optimized geometry.
845  S = self.simulation.context.getState(getPositions=True, getEnergy=True)
846  X1 = np.array([j for i, j in enumerate(S.getPositions().value_in_unit(angstrom)) if self.AtomMask[i]])
847  E = S.getPotentialEnergy().value_in_unit(kilocalorie_per_mole)
848  # Align to original geometry.
849  M = deepcopy(self.mol[0])
850  M.xyzs = [X0, X1]
851  if not self.pbc:
852  M.align(center=False)
853  X1 = M.xyzs[1]
854  # Set geometry in OpenMM, requires some hoops.
855  mod = Modeller(self.pdb.topology, [Vec3(i[0],i[1],i[2]) for i in X1]*angstrom)
856  mod.addExtraParticles(self.forcefield)
857  # self.simulation.context.setPositions(ResetVirtualSites(mod.getPositions(), self.system))
858  self.simulation.context.setPositions(ResetVirtualSites_fast(mod.getPositions(), self.vsinfo))
859  return E, M.ref_rmsd(0)[1]
