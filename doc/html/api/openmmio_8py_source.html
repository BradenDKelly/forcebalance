30 def get_dipole(simulation,q=None,positions=None):
31  """Return the current dipole moment in Debye.
32  Note that this quantity is meaningless if the system carries a net charge."""
33  dx = 0.0
34  dy = 0.0
35  dz = 0.0
36  enm_debye = 48.03204255928332 # Conversion factor from e*nm to Debye
37  for i in simulation.system.getForces():
38  if i.__class__.__name__ == "AmoebaMultipoleForce":
39  mm = i.getSystemMultipoleMoments(simulation.context)
40  dx += mm[1]
41  dy += mm[2]
42  dz += mm[3]
43  if i.__class__.__name__ == "NonbondedForce":
44  # Get array of charges.
45  if q == None:
46  q = np.array([i.getParticleParameters(j)[0]._value for j in range(i.getNumParticles())])
47  # Get array of positions in nanometers.
48  if positions == None:
49  positions = simulation.context.getState(getPositions=True).getPositions()
50  #x = np.array([j._value for j in positions])
51  x = np.array(positions.value_in_unit(nanometer))
52  # Multiply charges by positions to get dipole moment.
53  dip = enm_debye * np.sum(x*q.reshape(-1,1),axis=0)
54  dx += dip[0]
55  dy += dip[1]
56  dz += dip[2]
57  return [dx,dy,dz]
