36 def write_mdp(fout, options, fin=None, defaults={}, verbose=False):
37  """
38  Create or edit a Gromacs MDP file. The MDP file contains GROMACS run parameters.
39  @param[in] fout Output file name, can be the same as input file name.
40  @param[in] options Dictionary containing mdp options. Existing options are replaced, new options are added at the end.
41  @param[in] fin Input file name.
42  @param[in] defaults Default options to add to the mdp only if they don't already exist.
43  @param[in] verbose Print out all modifications to the file.
44  """
45  clashes = ["pbc"]
46  # Make sure that the keys are lowercase, and the values are all strings.
47  options = OrderedDict([(key.lower().replace('-','_'), str(val) if val != None else None) for key, val in options.items()])
48  # List of lines in the output file.
49  out = []
50  # List of options in the output file.
51  haveopts = []
52  if fin != None and os.path.isfile(fin):
53  for line in open(fin).readlines():
54  line = line.strip().expandtabs()
55  # The line structure should look something like this:
56  # key = value ; comments
57  # First split off the comments.
58  if len(line) == 0:
59  out.append('')
60  continue
61  s = line.split(';',1)
62  data = s[0]
63  comms = s[1] if len(s) > 1 else None
64  # Pure comment lines or empty lines get appended to the output.
65  if set(data).issubset([' ']):
66  out.append(line)
67  continue
68  # Now split off the key and value fields at the equals sign.
69  keyf, valf = data.split('=',1)
70  key = keyf.strip().lower().replace('-','_')
71  haveopts.append(key)
72  if key in options:
73  val = options[key]
74  val0 = valf.strip()
75  if key in clashes and val != val0:
76  raise RuntimeError("write_mdp tried to set %s = %s but its original value was %s = %s" % (key, val, key, val0))
77  # Passing None as the value causes the option to be deleted
78  if val == None: continue
79  if len(val) < len(valf):
80  valf = ' ' + val + ' '*(len(valf) - len(val)-1)
81  else:
82  valf = ' ' + val + ' '
83  lout = [keyf, '=', valf]
84  if comms != None:
85  lout += [';',comms]
86  out.append(''.join(lout))
87  else:
88  out.append(line)
89  for key, val in options.items():
90  key = key.lower().replace('-','_')
91  if key not in haveopts:
92  haveopts.append(key)
93  out.append("%-20s = %s" % (key, val))
94  # Fill in some default options.
95  for key, val in defaults.items():
96  key = key.lower().replace('-','_')
97  options[key] = val
98  if key not in haveopts:
99  out.append("%-20s = %s" % (key, val))
100  file_out = wopen(fout)
101  for line in out:
102  print >> file_out, line
103  if verbose:
104  printcool_dictionary(options, title="%s -> %s with options:" % (fin, fout))
105  file_out.close()

211 def parse_atomtype_line(line):
212  """ Parses the 'atomtype' line.
213 
214  Parses lines like this:\n
215  <tt> opls_135 CT 6 12.0107 0.0000 A 3.5000e-01 2.7614e-01\n
216  C 12.0107 0.0000 A 3.7500e-01 4.3932e-01\n
217  Na 11 22.9897 0.0000 A 6.068128070229e+03 2.662662556402e+01 0.0000e+00 ; PRM 5 6\n </tt>
218  Look at all the variety!
219 
220  @param[in] line Input line.
221  @return answer Dictionary containing:\n
222  atom type\n
223  bonded atom type (if any)\n
224  atomic number (if any)\n
225  atomic mass\n
226  charge\n
227  particle type\n
228  force field parameters\n
229  number of optional fields
230  """
231  # First split the line up to the comment. We don't care about the comment at this time
232  sline = line.split(';')[0].split()
233  # The line must contain at least six fields to be considered data.
234  if len(sline) < 6:
235  return
236  # Using variable "wrd" because the line has a variable number of fields
237  # Can you think of a better way?
238  wrd = 0
239  bonus = 0
240  atomtype = sline[wrd]
241  batomtype = sline[wrd]
242  wrd += 1
243  # The bonded atom type, a pecularity of OPLS-AA
244  # Test if it begins with a letter. Seems to work. :)
245  if re.match('[A-Za-z]',sline[wrd]):
246  batomtype = sline[wrd]
247  wrd += 1
248  bonus += 1
249  # Now to test if the next line is an atomic number or a mass.
250  # Atomic numbers never have decimals...
251  atomicnum = -1
252  if isint(sline[wrd]):
253  atomicnum = int(sline[wrd])
254  wrd += 1
255  bonus += 1
256  # The mass can be overridden in the 'atoms' section.
257  mass = float(sline[wrd])
258  wrd += 1
259  # Atom types have a default charge though this is almost always overridden
260  chg = float(sline[wrd])
261  wrd += 1
262  # Particle type. Actual atom or virtual site?
263  ptp = sline[wrd]
264  wrd += 1
265  param = [float(i) for i in sline[wrd:]]
266  answer = {'atomtype':atomtype, 'batomtype':batomtype, 'atomicnum':atomicnum, 'mass':mass, 'chg':chg, 'ptp':ptp, 'param':param, 'bonus':bonus}
267  return answer

269 class ITP_Reader(BaseReader):
270 
271  """Finite state machine for parsing GROMACS force field files.
272 
273  We open the force field file and read all of its lines. As we loop
274  through the force field file, we look for two types of tags: (1) section
275  markers, in GMX indicated by [ section_name ], which allows us to determine
276  the section, and (2) parameter tags, indicated by the 'PRM' or 'RPT' keywords.
277 
278  As we go through the file, we figure out the atoms involved in the interaction
279  described on each line.
280 
281  When a 'PRM' keyword is indicated, it is followed by a number which is the field
282  in the line to be modified, starting with zero. Based on the field number and the
283  section name, we can figure out the parameter type. With the parameter type
284  and the atoms in hand, we construct a 'parameter identifier' or pid which uniquely
285  identifies that parameter. We also store the physical parameter value in an array
286  called 'pvals0' and the precise location of that parameter (by filename, line number,
287  and field number) in a list called 'pfields'.
288 
289  An example: Suppose in 'my_ff.itp' I encounter the following on lines 146 and 147:
290 
291  @code
292  [ angletypes ]
293  CA CB O 1 109.47 350.00 ; PRM 4 5
294  @endcode
295 
296  From reading <tt>[ angletypes ]</tt> I know I'm in the 'angletypes' section.
297 
298  On the next line, I notice two parameters on fields 4 and 5.
299 
300  From the atom types, section type and field number I know the parameter IDs are <tt>'ANGLESBCACBO'</tt> and <tt>'ANGLESKCACBO'</tt>.
301 
302  After building <tt>map={'ANGLESBCACBO':1,'ANGLESKCACBO':2}</tt>, I store the values in
303  an array: <tt>pvals0=array([109.47,350.00])</tt>, and I put the parameter locations in
304  pfields: <tt>pfields=[['my_ff.itp',147,4,1.0],['my_ff.itp',146,5,1.0]]</tt>. The 1.0
305  is a 'multiplier' and I will explain it below.
306 
307  Note that in the creation of parameter IDs, we run into the issue that the atoms
308  involved in the interaction may be labeled in reverse order (e.g. <tt>OCACB</tt>). Thus,
309  we store both the normal and the reversed parameter ID in the map.
310 
311  Parameter repetition and multiplier:
312 
313  If <tt>'RPT'</tt> is encountered in the line, it is always in the syntax:
314  <tt>'RPT 4 ANGLESBCACAH 5 MINUS_ANGLESKCACAH /RPT'</tt>. In this case, field 4 is replaced by
315  the stored parameter value corresponding to <tt>ANGLESBCACAH</tt> and field 5 is replaced by
316  -1 times the stored value of <tt>ANGLESKCACAH</tt>. Now I just picked this as an example,
317  I don't think people actually want a negative angle force constant .. :) the <tt>MINUS</tt>
318  keyword does come in handy for assigning atomic charges and virtual site positions.
319  In order to achieve this, a multiplier of -1.0 is stored into pfields instead of 1.0.
320 
321  @todo Note that I can also create the opposite virtual site position by changing the atom
322  labeling, woo!
323

793  def evaluate_(self, force=False, dipole=False, traj=None):

796  Utility function for computing energy, and (optionally) forces and dipoles using GROMACS.
797
