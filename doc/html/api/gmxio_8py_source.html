99 def parse_atomtype_line(line):
100  """ Parses the 'atomtype' line.
101 
102  Parses lines like this:\n
103  <tt> opls_135 CT 6 12.0107 0.0000 A 3.5000e-01 2.7614e-01\n
104  C 12.0107 0.0000 A 3.7500e-01 4.3932e-01\n
105  Na 11 22.9897 0.0000 A 6.068128070229e+03 2.662662556402e+01 0.0000e+00 ; PARM 5 6\n </tt>
106  Look at all the variety!
107 
108  @param[in] line Input line.
109  @return answer Dictionary containing:\n
110  atom type\n
111  bonded atom type (if any)\n
112  atomic number (if any)\n
113  atomic mass\n
114  charge\n
115  particle type\n
116  force field parameters\n
117  number of optional fields
118  """
119  # First split the line up to the comment. We don't care about the comment at this time
120  sline = line.split(';')[0].split()
121  # The line must contain at least six fields to be considered data.
122  if len(sline) < 6:
123  return
124  # Using variable "wrd" because the line has a variable number of fields
125  # Can you think of a better way?
126  wrd = 0
127  bonus = 0
128  atomtype = sline[wrd]
129  batomtype = sline[wrd]
130  wrd += 1
131  # The bonded atom type, a pecularity of OPLS-AA
132  # Test if it begins with a letter. Seems to work. :)
133  if match('[A-Za-z]',sline[wrd]):
134  batomtype = sline[wrd]
135  wrd += 1
136  bonus += 1
137  # Now to test if the next line is an atomic number or a mass.
138  # Atomic numbers never have decimals...
139  atomicnum = -1
140  if isint(sline[wrd]):
141  atomicnum = int(sline[wrd])
142  wrd += 1
143  bonus += 1
144  # The mass can be overridden in the 'atoms' section.
145  mass = float(sline[wrd])
146  wrd += 1
147  # Atom types have a default charge though this is almost always overridden
148  chg = float(sline[wrd])
149  wrd += 1
150  # Particle type. Actual atom or virtual site?
151  ptp = sline[wrd]
152  wrd += 1
153  param = [float(i) for i in sline[wrd:]]
154  answer = {'atomtype':atomtype, 'batomtype':batomtype, 'atomicnum':atomicnum, 'mass':mass, 'chg':chg, 'ptp':ptp, 'param':param, 'bonus':bonus}
155  return answer

157 class ITP_Reader(BaseReader):
158 
159  """Finite state machine for parsing GROMACS force field files.
160 
161  We open the force field file and read all of its lines. As we loop
162  through the force field file, we look for two types of tags: (1) section
163  markers, in GMX indicated by [ section_name ], which allows us to determine
164  the section, and (2) parameter tags, indicated by the 'PARM' or 'RPT' keywords.
165 
166  As we go through the file, we figure out the atoms involved in the interaction
167  described on each line.
168 
169  When a 'PARM' keyword is indicated, it is followed by a number which is the field
170  in the line to be modified, starting with zero. Based on the field number and the
171  section name, we can figure out the parameter type. With the parameter type
172  and the atoms in hand, we construct a 'parameter identifier' or pid which uniquely
173  identifies that parameter. We also store the physical parameter value in an array
174  called 'pvals0' and the precise location of that parameter (by filename, line number,
175  and field number) in a list called 'pfields'.
176 
177  An example: Suppose in 'my_ff.itp' I encounter the following on lines 146 and 147:
178 
179  @code
180  [ angletypes ]
181  CA CB O 1 109.47 350.00 ; PARM 4 5
182  @endcode
183 
184  From reading <tt>[ angletypes ]</tt> I know I'm in the 'angletypes' section.
185 
186  On the next line, I notice two parameters on fields 4 and 5.
187 
188  From the atom types, section type and field number I know the parameter IDs are <tt>'ANGLESBCACBO'</tt> and <tt>'ANGLESKCACBO'</tt>.
189 
190  After building <tt>map={'ANGLESBCACBO':1,'ANGLESKCACBO':2}</tt>, I store the values in
191  an array: <tt>pvals0=array([109.47,350.00])</tt>, and I put the parameter locations in
192  pfields: <tt>pfields=[['my_ff.itp',147,4,1.0],['my_ff.itp',146,5,1.0]]</tt>. The 1.0
193  is a 'multiplier' and I will explain it below.
194 
195  Note that in the creation of parameter IDs, we run into the issue that the atoms
196  involved in the interaction may be labeled in reverse order (e.g. <tt>OCACB</tt>). Thus,
197  we store both the normal and the reversed parameter ID in the map.
198 
199  Parameter repetition and multiplier:
200 
201  If <tt>'RPT'</tt> is encountered in the line, it is always in the syntax:
202  <tt>'RPT 4 ANGLESBCACAH 5 MINUS_ANGLESKCACAH /RPT'</tt>. In this case, field 4 is replaced by
203  the stored parameter value corresponding to <tt>ANGLESBCACAH</tt> and field 5 is replaced by
204  -1 times the stored value of <tt>ANGLESKCACAH</tt>. Now I just picked this as an example,
205  I don't think people actually want a negative angle force constant .. :) the <tt>MINUS</tt>
206  keyword does come in handy for assigning atomic charges and virtual site positions.
207  In order to achieve this, a multiplier of -1.0 is stored into pfields instead of 1.0.
208 
209  @todo Note that I can also create the opposite virtual site position by changing the atom
210  labeling, woo!
211 
212  """

232  def feed(self, line):
233  """ Given a line, determine the interaction type and the atoms involved (the suffix).
234 
235  For example, we want \n
236  <tt> H O H 5 1.231258497536e+02 4.269161426840e+02 -1.033397697685e-02 1.304674117410e+04 ; PARM 4 5 6 7 </tt> \n
237  to give us itype = 'UREY_BRADLEY' and suffix = 'HOH'
238 
239  If we are in a TypeSection, it returns a list of atom types; \n
240  If we are in a TopolSection, it returns a list of atom names.
241 
242  The section is essentially a case statement that picks out the
243  appropriate interaction type and makes a list of the atoms
244  involved
245 
246  Note that we can call gmxdump for this as well, but I
247  prefer to read the force field file directly.
248 
249  ToDo: [ atoms ] section might need to be more flexible to accommodate optional fields
250 
251  """
252  s = line.split()
253  atom = []
254  self.itype = None
255  self.ln += 1
256  # No sense in doing anything for an empty line or a comment line.
257  if len(s) == 0 or match('^ *;',line): return None, None
258  # Now go through all the cases.
259  if match('^ *\[.*\]',line):
260  # Makes a word like "atoms", "bonds" etc.
261  self.sec = sub('[\[\] \n]','',line.strip())
262  elif self.sec == 'defaults':
263  self.itype = 'DEF'
264  self.nbtype = int(s[0])
265  elif self.sec == 'moleculetype':
266  self.mol = s[0]
267  elif self.sec == 'atomtypes':

372  def read_topology(self):
373  """ This function parses the GROMACS topology file (.top) which contains a listing of
374  the molecules in the simulation. For each molecule, it loads up a "FFMolecule" dictionary
375  which contains information about each atom in the molecule - which residue (i.e. molecular fragment)
376  the atom belongs in, the charge group, the particle type etc.
377 
378  This allows us to do things like condense the gradients into net forces and torques,
379  determine which particles are real atoms and which are virtual sites, and so on.
380  """
381  section = None
382  ResidueCounter = -1
383  ChargeGroupCounter = -1
384  MoleculeCounter = -1
385  for line in open(os.path.join(self.root, self.tgtdir, self.topfnm)).readlines():
386  s = line.split()
387  # No sense in doing anything for an empty line or a comment line.
388  if len(s) == 0 or match('^;',line): continue
389  # Now go through all the cases.
390  if match('^\[.*\]',line):
391  # Makes a word like "atoms", "bonds" etc.
392  section = sub('[\[\] \n]','',line)
393  elif section == 'molecules':
394  molname = s[0]
395  nummol = int(s[1])
396  FFMolecule = self.FF.FFMolecules[molname]
397  mollen = len(FFMolecule)
398  for i in range(nummol):
399  resnum = -1
400  cgnum = -1
401  MoleculeCounter += 1
402  for j in FFMolecule:
403  if j['ResidueNumber'] != resnum:
404  resnum = j['ResidueNumber']
405  ResidueCounter += 1
406  if j['ChargeGroupNumber'] != cgnum:
407  cgnum = j['ChargeGroupNumber']
408  ChargeGroupCounter += 1
409  self.AtomLists['ResidueNumber'].append(ResidueCounter)
410  self.AtomLists['MoleculeNumber'].append(MoleculeCounter)
411  self.AtomLists['ChargeGroupNumber'].append(ChargeGroupCounter)
412  self.AtomLists['ParticleType'].append(j['ParticleType'])
413  self.AtomLists['Mass'].append(j['Mass'])
414  self.AtomMask = array([i=='A' for i in self.AtomLists['ParticleType']])
415  self.topology_flag = True

551  if dielectric:

566  Diel_A = QChem_Dielectric_Energy("qchem_monoA.in",wq)
