00029 def edit_mdp(fin, fout, options, verbose=False):
00030     """
00031     Create or edit a Gromacs MDP file.
00032     @param[in] fin Input file name.
00033     @param[in] fout Output file name, can be the same as input file name.
00034     @param[in] options Dictionary containing mdp options. Existing options are replaced, new options are added at the end.
00035     """
00036     # Make sure that the keys are lowercase, and the values are all strings.
00037     options = OrderedDict([(key.lower(), str(val)) for key, val in options.items()])
00038     # List of lines in the output file.
00039     out = []
00040     # List of options in the output file.
00041     haveopts = []
00042     if os.path.isfile(fin):
00043         for line in open(fin).readlines():
00044             line    = line.strip().expandtabs()
00045             # The line structure should look something like this:
00046             # key   = value  ; comments
00047             # First split off the comments.
00048             if len(line) == 0:
00049                 out.append('')
00050             s = line.split(';',1)
00051             data = s[0]
00052             comms = s[1] if len(s) > 1 else None
00053             # Pure comment lines or empty lines get appended to the output.
00054             if set(data).issubset([' ']):
00055                 out.append(line)
00056                 continue
00057             # Now split off the key and value fields at the equals sign.
00058             keyf, valf = data.split('=',1)
00059             key = keyf.strip().lower()
00060             valen = len(valf)
00061             if key in options:
00062                 val = options[key]
00063                 if len(val) < len(valf):
00064                     valf = ' ' + val + ' '*(len(valf) - len(val)-1)
00065                 else:
00066                     valf = ' ' + val + ' '
00067                 lout = [keyf, '=', valf]
00068                 if comms != None:
00069                     lout += [';',comms]
00070                 out.append(''.join(lout))
00071                 haveopts.append(key)
00072             else:
00073                 out.append(line)
00074     for key, val in options.items():
00075         if key not in haveopts:
00076             out.append("%-20s = %s" % (key, val))
00077     file_out = open(fout,'w')
00078     for line in out:
00079         print >> file_out, line
00080     if verbose:
00081         printcool_dictionary(options, title="%s -> %s with options:" % (fin, fout))
00082     file_out.close()

00161 def parse_atomtype_line(line):
00162     """ Parses the 'atomtype' line.
00163     
00164     Parses lines like this:\n
00165     <tt> opls_135     CT    6   12.0107    0.0000    A    3.5000e-01    2.7614e-01\n
00166     C       12.0107    0.0000    A    3.7500e-01    4.3932e-01\n
00167     Na  11    22.9897    0.0000    A    6.068128070229e+03  2.662662556402e+01  0.0000e+00 ; PARM 5 6\n </tt>
00168     Look at all the variety!
00169 
00170     @param[in] line Input line.
00171     @return answer Dictionary containing:\n
00172     atom type\n
00173     bonded atom type (if any)\n
00174     atomic number (if any)\n
00175     atomic mass\n
00176     charge\n
00177     particle type\n
00178     force field parameters\n
00179     number of optional fields
00180     """
00181     # First split the line up to the comment.  We don't care about the comment at this time
00182     sline = line.split(';')[0].split()
00183     # The line must contain at least six fields to be considered data.
00184     if len(sline) < 6:
00185         return
00186     # Using variable "wrd" because the line has a variable number of fields
00187     # Can you think of a better way?
00188     wrd = 0
00189     bonus = 0
00190     atomtype = sline[wrd]
00191     batomtype = sline[wrd]
00192     wrd += 1
00193     # The bonded atom type, a pecularity of OPLS-AA
00194     # Test if it begins with a letter.  Seems to work. :)
00195     if re.match('[A-Za-z]',sline[wrd]):
00196         batomtype = sline[wrd]
00197         wrd += 1
00198         bonus += 1
00199     # Now to test if the next line is an atomic number or a mass.
00200     # Atomic numbers never have decimals...
00201     atomicnum = -1
00202     if isint(sline[wrd]):
00203         atomicnum = int(sline[wrd])
00204         wrd += 1
00205         bonus += 1
00206     # The mass can be overridden in the 'atoms' section.
00207     mass = float(sline[wrd])
00208     wrd += 1
00209     # Atom types have a default charge though this is almost always overridden
00210     chg  = float(sline[wrd])
00211     wrd += 1
00212     # Particle type. Actual atom or virtual site?
00213     ptp  = sline[wrd]
00214     wrd += 1
00215     param = [float(i) for i in sline[wrd:]]
00216     answer = {'atomtype':atomtype, 'batomtype':batomtype, 'atomicnum':atomicnum, 'mass':mass, 'chg':chg, 'ptp':ptp, 'param':param, 'bonus':bonus}
00217     return answer

00221     """Finite state machine for parsing GROMACS force field files.
00222     
00223     We open the force field file and read all of its lines.  As we loop
00224     through the force field file, we look for two types of tags: (1) section
00225     markers, in GMX indicated by [ section_name ], which allows us to determine
00226     the section, and (2) parameter tags, indicated by the 'PARM' or 'RPT' keywords.
00227     
00228     As we go through the file, we figure out the atoms involved in the interaction
00229     described on each line.
00230     
00231     When a 'PARM' keyword is indicated, it is followed by a number which is the field
00232     in the line to be modified, starting with zero.  Based on the field number and the
00233     section name, we can figure out the parameter type.  With the parameter type
00234     and the atoms in hand, we construct a 'parameter identifier' or pid which uniquely
00235     identifies that parameter.  We also store the physical parameter value in an array
00236     called 'pvals0' and the precise location of that parameter (by filename, line number,
00237     and field number) in a list called 'pfields'.
00238     
00239     An example: Suppose in 'my_ff.itp' I encounter the following on lines 146 and 147:
00240     
00241     @code
00242     [ angletypes ]
00243     CA   CB   O   1   109.47  350.00  ; PARM 4 5
00244     @endcode
00245     
00246     From reading <tt>[ angletypes ]</tt> I know I'm in the 'angletypes' section.
00247     
00248     On the next line, I notice two parameters on fields 4 and 5.
00249     
00250     From the atom types, section type and field number I know the parameter IDs are <tt>'ANGLESBCACBO'</tt> and <tt>'ANGLESKCACBO'</tt>.
00251     
00252     After building <tt>map={'ANGLESBCACBO':1,'ANGLESKCACBO':2}</tt>, I store the values in
00253     an array: <tt>pvals0=array([109.47,350.00])</tt>, and I put the parameter locations in
00254     pfields: <tt>pfields=[['my_ff.itp',147,4,1.0],['my_ff.itp',146,5,1.0]]</tt>.  The 1.0
00255     is a 'multiplier' and I will explain it below.
00256     
00257     Note that in the creation of parameter IDs, we run into the issue that the atoms
00258     involved in the interaction may be labeled in reverse order (e.g. <tt>OCACB</tt>).  Thus,
00259     we store both the normal and the reversed parameter ID in the map.
00260     
00261     Parameter repetition and multiplier:
00262     
00263     If <tt>'RPT'</tt> is encountered in the line, it is always in the syntax:
00264     <tt>'RPT 4 ANGLESBCACAH 5 MINUS_ANGLESKCACAH /RPT'</tt>.  In this case, field 4 is replaced by
00265     the stored parameter value corresponding to <tt>ANGLESBCACAH</tt> and field 5 is replaced by
00266     -1 times the stored value of <tt>ANGLESKCACAH</tt>.  Now I just picked this as an example,
00267     I don't think people actually want a negative angle force constant .. :) the <tt>MINUS</tt>
00268     keyword does come in handy for assigning atomic charges and virtual site positions.
00269     In order to achieve this, a multiplier of -1.0 is stored into pfields instead of 1.0.
00270     
00271     @todo Note that I can also create the opposite virtual site position by changing the atom
00272     labeling, woo!
00273     
00274     """
