00024 def edit_mdp(fin, fout, options, verbose=False):
00025     """
00026     Create or edit a Gromacs MDP file.
00027     @param[in] fin Input file name.
00028     @param[in] fout Output file name, can be the same as input file name.
00029     @param[in] options Dictionary containing mdp options. Existing options are replaced, new options are added at the end.
00030     """
00031     # Make sure that the keys are lowercase, and the values are all strings.
00032     options = OrderedDict([(key.lower(), str(val)) for key, val in options.items()])
00033     # List of lines in the output file.
00034     out = []
00035     # List of options in the output file.
00036     haveopts = []
00037     if os.path.isfile(fin):
00038         for line in open(fin).readlines():
00039             line    = line.strip().expandtabs()
00040             # The line structure should look something like this:
00041             # key   = value  ; comments
00042             # First split off the comments.
00043             if len(line) == 0:
00044                 out.append('')
00045             s = line.split(';',1)
00046             data = s[0]
00047             comms = s[1] if len(s) > 1 else None
00048             # Pure comment lines or empty lines get appended to the output.
00049             if set(data).issubset([' ']):
00050                 out.append(line)
00051                 continue
00052             # Now split off the key and value fields at the equals sign.
00053             keyf, valf = data.split('=',1)
00054             key = keyf.strip().lower()
00055             valen = len(valf)
00056             if key in options:
00057                 val = options[key]
00058                 if len(val) < len(valf):
00059                     valf = ' ' + val + ' '*(len(valf) - len(val)-1)
00060                 else:
00061                     valf = ' ' + val + ' '
00062                 lout = [keyf, '=', valf]
00063                 if comms != None:
00064                     lout += [';',comms]
00065                 out.append(''.join(lout))
00066                 haveopts.append(key)
00067             else:
00068                 out.append(line)
00069     for key, val in options.items():
00070         if key not in haveopts:
00071             out.append("%-20s = %s" % (key, val))
00072     file_out = open(fout,'w')
00073     for line in out:
00074         print >> file_out, line
00075     if verbose:
00076         printcool_dictionary(options, title="%s -> %s with options:" % (fin, fout))
00077     file_out.close()

00156 def parse_atomtype_line(line):
00157     """ Parses the 'atomtype' line.
00158     
00159     Parses lines like this:\n
00160     <tt> opls_135     CT    6   12.0107    0.0000    A    3.5000e-01    2.7614e-01\n
00161     C       12.0107    0.0000    A    3.7500e-01    4.3932e-01\n
00162     Na  11    22.9897    0.0000    A    6.068128070229e+03  2.662662556402e+01  0.0000e+00 ; PARM 5 6\n </tt>
00163     Look at all the variety!
00164 
00165     @param[in] line Input line.
00166     @return answer Dictionary containing:\n
00167     atom type\n
00168     bonded atom type (if any)\n
00169     atomic number (if any)\n
00170     atomic mass\n
00171     charge\n
00172     particle type\n
00173     force field parameters\n
00174     number of optional fields
00175     """
00176     # First split the line up to the comment.  We don't care about the comment at this time
00177     sline = line.split(';')[0].split()
00178     # The line must contain at least six fields to be considered data.
00179     if len(sline) < 6:
00180         return
00181     # Using variable "wrd" because the line has a variable number of fields
00182     # Can you think of a better way?
00183     wrd = 0
00184     bonus = 0
00185     atomtype = sline[wrd]
00186     batomtype = sline[wrd]
00187     wrd += 1
00188     # The bonded atom type, a pecularity of OPLS-AA
00189     # Test if it begins with a letter.  Seems to work. :)
00190     if re.match('[A-Za-z]',sline[wrd]):
00191         batomtype = sline[wrd]
00192         wrd += 1
00193         bonus += 1
00194     # Now to test if the next line is an atomic number or a mass.
00195     # Atomic numbers never have decimals...
00196     atomicnum = -1
00197     if isint(sline[wrd]):
00198         atomicnum = int(sline[wrd])
00199         wrd += 1
00200         bonus += 1
00201     # The mass can be overridden in the 'atoms' section.
00202     mass = float(sline[wrd])
00203     wrd += 1
00204     # Atom types have a default charge though this is almost always overridden
00205     chg  = float(sline[wrd])
00206     wrd += 1
00207     # Particle type. Actual atom or virtual site?
00208     ptp  = sline[wrd]
00209     wrd += 1
00210     param = [float(i) for i in sline[wrd:]]
00211     answer = {'atomtype':atomtype, 'batomtype':batomtype, 'atomicnum':atomicnum, 'mass':mass, 'chg':chg, 'ptp':ptp, 'param':param, 'bonus':bonus}
00212     return answer

00214 class ITP_Reader(BaseReader):

00216     """Finite state machine for parsing GROMACS force field files.
00217     
00218     We open the force field file and read all of its lines.  As we loop
00219     through the force field file, we look for two types of tags: (1) section
00220     markers, in GMX indicated by [ section_name ], which allows us to determine
00221     the section, and (2) parameter tags, indicated by the 'PARM' or 'RPT' keywords.
00222     
00223     As we go through the file, we figure out the atoms involved in the interaction
00224     described on each line.
00225     
00226     When a 'PARM' keyword is indicated, it is followed by a number which is the field
00227     in the line to be modified, starting with zero.  Based on the field number and the
00228     section name, we can figure out the parameter type.  With the parameter type
00229     and the atoms in hand, we construct a 'parameter identifier' or pid which uniquely
00230     identifies that parameter.  We also store the physical parameter value in an array
00231     called 'pvals0' and the precise location of that parameter (by filename, line number,
00232     and field number) in a list called 'pfields'.
00233     
00234     An example: Suppose in 'my_ff.itp' I encounter the following on lines 146 and 147:
00235     
00236     @code
00237     [ angletypes ]
00238     CA   CB   O   1   109.47  350.00  ; PARM 4 5
00239     @endcode
00240     
00241     From reading <tt>[ angletypes ]</tt> I know I'm in the 'angletypes' section.
00242     
00243     On the next line, I notice two parameters on fields 4 and 5.
00244     
00245     From the atom types, section type and field number I know the parameter IDs are <tt>'ANGLESBCACBO'</tt> and <tt>'ANGLESKCACBO'</tt>.
00246     
00247     After building <tt>map={'ANGLESBCACBO':1,'ANGLESKCACBO':2}</tt>, I store the values in
00248     an array: <tt>pvals0=array([109.47,350.00])</tt>, and I put the parameter locations in
00249     pfields: <tt>pfields=[['my_ff.itp',147,4,1.0],['my_ff.itp',146,5,1.0]]</tt>.  The 1.0
00250     is a 'multiplier' and I will explain it below.
00251     
00252     Note that in the creation of parameter IDs, we run into the issue that the atoms
00253     involved in the interaction may be labeled in reverse order (e.g. <tt>OCACB</tt>).  Thus,
00254     we store both the normal and the reversed parameter ID in the map.
00255     
00256     Parameter repetition and multiplier:
00257     
00258     If <tt>'RPT'</tt> is encountered in the line, it is always in the syntax:
00259     <tt>'RPT 4 ANGLESBCACAH 5 MINUS_ANGLESKCACAH /RPT'</tt>.  In this case, field 4 is replaced by
00260     the stored parameter value corresponding to <tt>ANGLESBCACAH</tt> and field 5 is replaced by
00261     -1 times the stored value of <tt>ANGLESKCACAH</tt>.  Now I just picked this as an example,
00262     I don't think people actually want a negative angle force constant .. :) the <tt>MINUS</tt>
00263     keyword does come in handy for assigning atomic charges and virtual site positions.
00264     In order to achieve this, a multiplier of -1.0 is stored into pfields instead of 1.0.
00265     
00266     @todo Note that I can also create the opposite virtual site position by changing the atom
00267     labeling, woo!
00268     
00269     """
