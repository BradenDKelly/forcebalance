290 def invert_svd(X,thresh=1e-12):
291 
292  """
293 
294  Invert a matrix using singular value decomposition.
295  @param[in] X The matrix to be inverted
296  @param[in] thresh The SVD threshold; eigenvalues below this are not inverted but set to zero
297  @return Xt The inverted matrix
298 
299  """
300 
301  u,s,vh = svd(X, full_matrices=0)
302  uh = mat(transpose(u))
303  v = mat(transpose(vh))
304  si = s.copy()
305  for i in range(s.shape[0]):
306  if abs(s[i]) > thresh:
307  si[i] = 1./s[i]
308  else:
309  si[i] = 0.0
310  si = mat(diag(si))
311  Xt = v*si*uh
312  return Xt

317 def get_least_squares(x, y, w = None, thresh=1e-12):
318  """
319  @code
320  __ __
321  | |
322  | 1 (x0) (x0)^2 (x0)^3 |
323  | 1 (x1) (x1)^2 (x1)^3 |
324  | 1 (x2) (x2)^2 (x2)^3 |
325  | 1 (x3) (x3)^2 (x3)^3 |
326  | 1 (x4) (x4)^2 (x4)^3 |
327  |__ __|
328 
329  @endcode
330 
331  @param[in] X (2-D array) An array of X-values (see above)
332  @param[in] Y (array) An array of Y-values (only used in getting the least squares coefficients)
333  @param[in] w (array) An array of weights, hopefully normalized to one.
334  @param[out] Beta The least-squares coefficients
335  @param[out] Hat The hat matrix that takes linear combinations of data y-values to give fitted y-values (weights)
336  @param[out] yfit The fitted y-values
337  @param[out] MPPI The Moore-Penrose pseudoinverse (multiply by Y to get least-squares coefficients, multiply by dY/dk to get derivatives of least-squares coefficients)
338  """
339  # X is a 'tall' matrix.
340  X = mat(x)
341  Y = col(y)
342  n_x = X.shape[0]
343  n_fit = X.shape[1]
344  if n_fit > n_x:
345  print "Argh? It seems like this problem is underdetermined!"
346  # Build the weight matrix.
347  if w != None:
348  if len(w) != n_x:
349  warn_press_key("The weight array length (%i) must be the same as the number of 'X' data points (%i)!" % len(w), n_x)
350  w /= mean(w)
351  WH = mat(diag(w**0.5))
352  else:
353  WH = mat(eye(n_x))
354  # Make the Moore-Penrose Pseudoinverse.
355  # if n_fit == n_x:
356  # MPPI = np.linalg.inv(WH*X)
357  # else:
358  # This resembles the formula (X'WX)^-1 X' W^1/2
359  MPPI = np.linalg.pinv(WH*X)
360  Beta = MPPI * WH * Y
361  Hat = WH * X * MPPI
362  yfit = flat(Hat * Y)
363  # Return three things: the least-squares coefficients, the hat matrix (turns y into yfit), and yfit
364  # We could get these all from MPPI, but I might get confused later on, so might as well do it here :P
365  return Beta, Hat, yfit, MPPI
