310 def monotonic(arr, start, end):
311  # Make sure an array is monotonically decreasing from the start to the end.
312  a0 = arr[start]
313  i0 = start
314  if end > start:
315  i = start+1
316  while i < end:
317  if arr[i] < a0:
318  arr[i0:i+1] = np.linspace(a0, arr[i], i-i0+1)
319  a0 = arr[i]
320  i0 = i
321  i += 1
322  if end < start:
323  i = start-1
324  while i >= end:
325  if arr[i] < a0:
326  arr[i:i0+1] = np.linspace(arr[i], a0, i0-i+1)
327  a0 = arr[i]
328  i0 = i

348 
349  Invert a matrix using singular value decomposition.
350  @param[in] X The matrix to be inverted
351  @param[in] thresh The SVD threshold; eigenvalues below this are not inverted but set to zero
352  @return Xt The inverted matrix
353

374  @code

376  | |
377  | 1 (x0) (x0)^2 (x0)^3 |
378  | 1 (x1) (x1)^2 (x1)^3 |
379  | 1 (x2) (x2)^2 (x2)^3 |

385 
386  @param[in] X (2-D array) An array of X-values (see above)

388  @param[in] w (array) An array of weights, hopefully normalized to one.
389  @param[out] Beta The least-squares coefficients

391  @param[out] yfit The fitted y-values
392  @param[out] MPPI The Moore-Penrose pseudoinverse (multiply by Y to get least-squares coefficients, multiply by dY/dk to get derivatives of least-squares coefficients)
393  """
394  # X is a 'tall' matrix.
395  X = np.matrix(x)
396  Y = col(y)

406  WH = np.matrix(np.diag(w**0.5))
407  else:
408  WH = np.matrix(np.eye(n_x))

425 def statisticalInefficiency(A_n, B_n=None, fast=False, mintime=3, warn=True):
426 
427  """
428  Compute the (cross) statistical inefficiency of (two) timeseries.
429 
430  Notes
431  The same timeseries can be used for both A_n and B_n to get the autocorrelation statistical inefficiency.
432  The fast method described in Ref [1] is used to compute g.
433 
434  References
435  [1] J. D. Chodera, W. C. Swope, J. W. Pitera, C. Seok, and K. A. Dill. Use of the weighted
436  histogram analysis method for the analysis of simulated and parallel tempering simulations.
437  JCTC 3(1):26-41, 2007.
438 
439  Examples
440 
441  Compute statistical inefficiency of timeseries data with known correlation time.
442 
443  >>> import timeseries
444  >>> A_n = timeseries.generateCorrelatedTimeseries(N=100000, tau=5.0)
445  >>> g = statisticalInefficiency(A_n, fast=True)
446 
447  @param[in] A_n (required, numpy array) - A_n[n] is nth value of
448  timeseries A. Length is deduced from vector.
449 
450  @param[in] B_n (optional, numpy array) - B_n[n] is nth value of
451  timeseries B. Length is deduced from vector. If supplied, the
452  cross-correlation of timeseries A and B will be estimated instead of
453  the autocorrelation of timeseries A.
454 
455  @param[in] fast (optional, boolean) - if True, will use faster (but
456  less accurate) method to estimate correlation time, described in
457  Ref. [1] (default: False)
458 
459  @param[in] mintime (optional, int) - minimum amount of correlation
460  function to compute (default: 3) The algorithm terminates after
461  computing the correlation time out to mintime when the correlation
462  function furst goes negative. Note that this time may need to be
463  increased if there is a strong initial negative peak in the
464  correlation function.
465 
466  @return g The estimated statistical inefficiency (equal to 1 + 2
467  tau, where tau is the correlation time). We enforce g >= 1.0.
468 
469  """
470  # Create numpy copies of input arguments.
471  A_n = np.array(A_n)
472  if B_n is not None:
473  B_n = np.array(B_n)
474  else:
475  B_n = np.array(A_n)
476  # Get the length of the timeseries.
477  N = A_n.shape[0]
478  # Be sure A_n and B_n have the same dimensions.
479  if(A_n.shape != B_n.shape):
480  raise ParameterError('A_n and B_n must have same dimensions.')
481  # Initialize statistical inefficiency estimate with uncorrelated value.
482  g = 1.0
483  # Compute mean of each timeseries.
484  mu_A = A_n.mean()
485  mu_B = B_n.mean()
486  # Make temporary copies of fluctuation from mean.
487  dA_n = A_n.astype(np.float64) - mu_A
488  dB_n = B_n.astype(np.float64) - mu_B
489  # Compute estimator of covariance of (A,B) using estimator that will ensure C(0) = 1.
490  sigma2_AB = (dA_n * dB_n).mean() # standard estimator to ensure C(0) = 1
491  # Trap the case where this covariance is zero, and we cannot proceed.
492  if(sigma2_AB == 0):
493  if warn:
494  logger.warning('Sample covariance sigma_AB^2 = 0 -- cannot compute statistical inefficiency\n')
495  return 1.0
496  # Accumulate the integrated correlation time by computing the normalized correlation time at
497  # increasing values of t. Stop accumulating if the correlation function goes negative, since
498  # this is unlikely to occur unless the correlation function has decayed to the point where it
499  # is dominated by noise and indistinguishable from zero.
500  t = 1
501  increment = 1
502  while (t < N-1):
503  # compute normalized fluctuation correlation function at time t
504  C = sum( dA_n[0:(N-t)]*dB_n[t:N] + dB_n[0:(N-t)]*dA_n[t:N] ) / (2.0 * float(N-t) * sigma2_AB)
505  # Terminate if the correlation function has crossed zero and we've computed the correlation
506  # function at least out to 'mintime'.
507  if (C <= 0.0) and (t > mintime):
508  break
509  # Accumulate contribution to the statistical inefficiency.
510  g += 2.0 * C * (1.0 - float(t)/float(N)) * float(increment)
511  # Increment t and the amount by which we increment t.
512  t += increment
513  # Increase the interval if "fast mode" is on.
514  if fast: increment += 1
515  # g must be at least unity
516  if (g < 1.0): g = 1.0
517  # Return the computed statistical inefficiency.
518  return g

940 def _exec(command, print_to_screen = False, outfnm = None, logfnm = None, stdin = "", print_command = True, copy_stdout = True, copy_stderr = False, persist = False, expand_cr=False, print_error=True, rbytes=1, **kwargs):
941  """Runs command line using subprocess, optionally returning stdout.
942  Options:
943  command (required) = Name of the command you want to execute
944  outfnm (optional) = Name of the output file name (overwritten if exists)
945  logfnm (optional) = Name of the log file name (appended if exists)
946  stdin (optional) = A string to be passed to stdin, as if it were typed (use newline character to mimic Enter key)
947  print_command = Whether to print the command.
948  copy_stdout = Copy the stdout stream; can set to False in strange situations
949  copy_stderr = Copy the stderr stream to the stdout stream; useful for GROMACS which prints out everything to stderr (argh.)
950  expand_cr = Whether to expand carriage returns into newlines (useful for GROMACS mdrun).
951  print_error = Whether to print error messages on a crash. Should be true most of the time.
952  persist = Continue execution even if the command gives a nonzero return code.
953  rbytes = Number of bytes to read from stdout and stderr streams at a time. GMX requires rbytes = 1 otherwise streams are interleaved. Higher values for speed.
954  """
955 
956  # Dictionary of options to be passed to the Popen object.
957  cmd_options={'shell':(type(command) is str), 'stdin':PIPE, 'stdout':PIPE, 'stderr':PIPE, 'universal_newlines':expand_cr}
958 
959  # "write to file" : Function for writing some characters to the log and/or output files.
960  def wtf(out):
961  if logfnm != None:
962  with open(logfnm,'a+') as f:
963  f.write(out)
964  f.flush()
965  if outfnm != None:
966  with open(outfnm,'w+' if wtf.first else 'a+') as f:
967  f.write(out)
968  f.flush()
969  wtf.first = False
970  wtf.first = True
971 
972  # Preserve backwards compatibility; sometimes None gets passed to stdin.
973  if stdin == None: stdin = ""
974 
975  if print_command:
976  logger.info("Executing process: \x1b[92m%-50s\x1b[0m%s%s%s\n" % (' '.join(command) if type(command) is list else command,
977  " Output: %s" % outfnm if outfnm != None else "",
978  " Append: %s" % logfnm if logfnm != None else "",
979  (" Stdin: %s" % stdin.replace('\n','\\n')) if stdin else ""))
980  wtf("Executing process: %s%s\n" % (command, (" Stdin: %s" % stdin.replace('\n','\\n')) if stdin else ""))
981 
982  cmd_options.update(kwargs)
983  p = subprocess.Popen(command, **cmd_options)
984 
985  # Write the stdin stream to the process.
986  p.stdin.write(stdin)
987  p.stdin.close()
988 
989  #===============================================================#
990  #| Read the output streams from the process. This is a bit |#
991  #| complicated because programs like GROMACS tend to print out |#
992  #| stdout as well as stderr streams, and also carriage returns |#
993  #| along with newline characters. |#
994  #===============================================================#
995  # stdout and stderr streams of the process.
996  streams = [p.stdout, p.stderr]
997  # These are functions that take chunks of lines (read) as inputs.
998  def process_out(read):
999  if print_to_screen: sys.stdout.write(read)
1000  if copy_stdout:
1001  process_out.stdout.append(read)
1002  wtf(read)
1003  process_out.stdout = []
1004 
1005  def process_err(read):
1006  if print_to_screen: sys.stderr.write(read)
1007  process_err.stderr.append(read)
1008  if copy_stderr:
1009  process_out.stdout.append(read)
1010  wtf(read)
1011  process_err.stderr = []
1012  # This reads the streams one byte at a time, and passes it to the LineChunker
1013  # which splits it by either newline or carriage return.
1014  # If the stream has ended, then it is removed from the list.
1015  with LineChunker(process_out) as out_chunker, LineChunker(process_err) as err_chunker:
1016  while True:
1017  to_read, _, _ = select(streams, [], [])
1018  for fh in to_read:
1019  if fh is p.stdout:
1020  read = fh.read(rbytes)
1021  if not read:
1022  streams.remove(p.stdout)
1023  p.stdout.close()
1024  else: out_chunker.push(read)
1025  elif fh is p.stderr:
1026  read = fh.read(rbytes)
1027  if not read:
1028  streams.remove(p.stderr)
1029  p.stderr.close()
1030  else: err_chunker.push(read)
1031  else:
1032  raise RuntimeError
1033  if len(streams) == 0: break
1034 
1035  p.wait()
1036 
1037  process_out.stdout = ''.join(process_out.stdout)
1038  process_err.stderr = ''.join(process_err.stderr)
1039 
1040  if p.returncode != 0:
1041  if process_err.stderr and print_error:
1042  logger.warning("Received an error message:\n")
1043  logger.warning("\n[====] \x1b[91mError Message\x1b[0m [====]\n")
1044  logger.warning(process_err.stderr)
1045  logger.warning("[====] \x1b[91mEnd o'Message\x1b[0m [====]\n")
1046  if persist:
1047  if print_error:
1048  logger.info("%s gave a return code of %i (it may have crashed) -- carrying on\n" % (command, p.returncode))
1049  else:
1050  # This code (commented out) would not throw an exception, but instead exit with the returncode of the crashed program.
1051  # sys.stderr.write("\x1b[1;94m%s\x1b[0m gave a return code of %i (\x1b[91mit may have crashed\x1b[0m)\n" % (command, p.returncode))
1052  # sys.exit(p.returncode)
1053  raise Exception("\x1b[1;94m%s\x1b[0m gave a return code of %i (\x1b[91mit may have crashed\x1b[0m)\n" % (command, p.returncode))
1054
