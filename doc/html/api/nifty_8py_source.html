00297 
00298     Invert a matrix using singular value decomposition. 
00299     @param[in] X The matrix to be inverted
00300     @param[in] thresh The SVD threshold; eigenvalues below this are not inverted but set to zero
00301     @return Xt The inverted matrix

00303     """

00305     u,s,vh = svd(X, full_matrices=0)
00306     uh     = mat(transpose(u))
00307     v      = mat(transpose(vh))

00309     for i in range(s.shape[0]):
00310         if abs(s[i]) > thresh:

00313             si[i] = 0.0

00323     @code

00327     | 1 (x1) (x1)^2 (x1)^3 |

00374 def statisticalInefficiency(A_n, B_n=None, fast=False, mintime=3, warn=True):
00375 
00376     """
00377     Compute the (cross) statistical inefficiency of (two) timeseries.
00378 
00379     Notes
00380       The same timeseries can be used for both A_n and B_n to get the autocorrelation statistical inefficiency.
00381       The fast method described in Ref [1] is used to compute g.
00382 
00383     References
00384       [1] J. D. Chodera, W. C. Swope, J. W. Pitera, C. Seok, and K. A. Dill. Use of the weighted
00385       histogram analysis method for the analysis of simulated and parallel tempering simulations.
00386       JCTC 3(1):26-41, 2007.
00387 
00388     Examples
00389 
00390     Compute statistical inefficiency of timeseries data with known correlation time.
00391 
00392     >>> import timeseries
00393     >>> A_n = timeseries.generateCorrelatedTimeseries(N=100000, tau=5.0)
00394     >>> g = statisticalInefficiency(A_n, fast=True)
00395 
00396     @param[in] A_n (required, numpy array) - A_n[n] is nth value of
00397     timeseries A.  Length is deduced from vector.
00398 
00399     @param[in] B_n (optional, numpy array) - B_n[n] is nth value of
00400     timeseries B.  Length is deduced from vector.  If supplied, the
00401     cross-correlation of timeseries A and B will be estimated instead of
00402     the autocorrelation of timeseries A.
00403 
00404     @param[in] fast (optional, boolean) - if True, will use faster (but
00405     less accurate) method to estimate correlation time, described in
00406     Ref. [1] (default: False)
00407 
00408     @param[in] mintime (optional, int) - minimum amount of correlation
00409     function to compute (default: 3) The algorithm terminates after
00410     computing the correlation time out to mintime when the correlation
00411     function furst goes negative.  Note that this time may need to be
00412     increased if there is a strong initial negative peak in the
00413     correlation function.
00414 
00415     @return g The estimated statistical inefficiency (equal to 1 + 2
00416     tau, where tau is the correlation time).  We enforce g >= 1.0.
00417 
00418     """
00419 
00420     # Create numpy copies of input arguments.
00421     A_n = array(A_n)
00422     if B_n is not None:
00423         B_n = array(B_n)
00424     else:
00425         B_n = array(A_n)
00426     # Get the length of the timeseries.
00427     N = A_n.size
00428     # Be sure A_n and B_n have the same dimensions.
00429     if(A_n.shape != B_n.shape):
00430         raise ParameterError('A_n and B_n must have same dimensions.')
00431     # Initialize statistical inefficiency estimate with uncorrelated value.
00432     g = 1.0
00433     # Compute mean of each timeseries.
00434     mu_A = A_n.mean()
00435     mu_B = B_n.mean()
00436     # Make temporary copies of fluctuation from mean.
00437     dA_n = A_n.astype(np.float64) - mu_A
00438     dB_n = B_n.astype(np.float64) - mu_B
00439     # Compute estimator of covariance of (A,B) using estimator that will ensure C(0) = 1.
00440     sigma2_AB = (dA_n * dB_n).mean() # standard estimator to ensure C(0) = 1
00441     # Trap the case where this covariance is zero, and we cannot proceed.
00442     if(sigma2_AB == 0):
00443         if warn:
00444             logger.warning('Sample covariance sigma_AB^2 = 0 -- cannot compute statistical inefficiency')
00445         return 1.0
00446     # Accumulate the integrated correlation time by computing the normalized correlation time at
00447     # increasing values of t.  Stop accumulating if the correlation function goes negative, since
00448     # this is unlikely to occur unless the correlation function has decayed to the point where it
00449     # is dominated by noise and indistinguishable from zero.
00450     t = 1
00451     increment = 1
00452     while (t < N-1):
00453         # compute normalized fluctuation correlation function at time t
00454         C = sum( dA_n[0:(N-t)]*dB_n[t:N] + dB_n[0:(N-t)]*dA_n[t:N] ) / (2.0 * float(N-t) * sigma2_AB)
00455         # Terminate if the correlation function has crossed zero and we've computed the correlation
00456         # function at least out to 'mintime'.
00457         if (C <= 0.0) and (t > mintime):
00458             break
00459         # Accumulate contribution to the statistical inefficiency.
00460         g += 2.0 * C * (1.0 - float(t)/float(N)) * float(increment)
00461         # Increment t and the amount by which we increment t.
00462         t += increment
00463         # Increase the interval if "fast mode" is on.
00464         if fast: increment += 1
00465     # g must be at least unity
00466     if (g < 1.0): g = 1.0
00467     # Return the computed statistical inefficiency.
00468     return g
