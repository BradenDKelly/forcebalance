18 class Target(forcebalance.BaseClass):
19 
20  """
21  Base class for all fitting targets.
22 
23  In ForceBalance a Target is defined as a set of reference data
24  plus a corresponding method to simulate that data using the force field.
25 
26  The 'computable quantities' may include energies and forces where the
27  reference values come from QM calculations (energy and force matching),
28  energies from an EDA analysis (Maybe in the future, FDA?), molecular
29  properties (like polarizability, refractive indices, multipole moments
30  or vibrational frequencies), relative entropies, and bulk properties.
31  Single-molecule or bulk properties can even come from the experiment!
32 
33  The central idea in ForceBalance is that each quantity makes a
34  contribution to the overall objective function. So we can build force
35  fields that fit several quantities at once, rather than putting all of
36  our chips behind energy and force matching. In the future
37  ForceBalance may even include multiobjective optimization into the
38  optimizer.
39 
40  The optimization is done by way of minimizing an 'objective
41  function', which is comprised of squared differences between the
42  computed and reference values. These differences are not computed
43  in this file, but rather in subclasses that use Target
44  as a base class. Thus, the contents of Target itself
45  are meant to be as general as possible, because the pertinent
46  variables apply to all types of fitting targets.
47 
48  An important node: Target requires that all subclasses
49  have a method get(self,mvals,AGrad=False,AHess=False)
50  that does the following:
51 
52  Inputs:
53  mvals = The parameter vector, which modifies the force field
54  (Note to self: We include mvals with each Target because we can create
55  copies of the force field and do finite difference derivatives)
56  AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
57 
58  Outputs:
59  Answer = {'X': Number, 'G': array(NP), 'H': array((NP,NP)) }
60  'X' = The objective function itself
61  'G' = The gradient, elements not computed analytically are zero
62  'H' = The Hessian, elements not computed analytically are zero
63 
64  This is the only global requirement of a Target.
65  Obviously 'get' itself is not defined here, because its
66  calculation will depend entirely on specifically which target
67  we wish to use. However, this should give us a unified framework
68  which will faciliate rapid implementation of Targets.
69 
70  Future work:
71  Robert suggested that I could enable automatic detection of which
72  parameters need to be computed by finite difference. Not a bad idea. :)
73 
74  """

249  def get_G(self,mvals=None):
250  """Computes the objective function contribution and its gradient.
251 
252  First the low-level 'get' method is called with the analytic gradient
253  switch turned on. Then we loop through the fd1_pids and compute
254  the corresponding elements of the gradient by finite difference,
255  if the 'fdgrad' switch is turned on. Alternately we can compute
256  the gradient elements and diagonal Hessian elements at the same time
257  using central difference if 'fdhessdiag' is turned on.
258 
259  In this function we also record which parameters cause a
260  nonzero change in the objective function contribution.
261  Parameters which do not change the objective function will
262  not be differentiated in subsequent calculations. This is
263  recorded in a text file in the targets directory.
264 
265  """
266  Ans = self.meta_get(mvals,1,0)
267  for i in self.pgrad:
268  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
269  if self.fdhessdiag:
270  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
271  elif self.fdgrad:
272  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
273  self.gct += 1
274  self.write_0grads(Ans)
275  return Ans

277  def get_H(self,mvals=None):
278  """Computes the objective function contribution and its gradient / Hessian.
279 
280  First the low-level 'get' method is called with the analytic gradient
281  and Hessian both turned on. Then we loop through the fd1_pids and compute
282  the corresponding elements of the gradient by finite difference,
283  if the 'fdgrad' switch is turned on.
284 
285  This is followed by looping through the fd2_pids and computing the corresponding
286  Hessian elements by finite difference. Forward finite difference is used
287  throughout for the sake of speed.
288  """
289  Ans = self.meta_get(mvals,1,1)
290  if self.fdhess:
291  for i in self.pgrad:
292  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
293  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
294  for i in self.pgrad:
295  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
296  FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
297  Ans['H'][i,:] = FDSlice
298  Ans['H'][:,i] = FDSlice
299  elif self.fdhessdiag:
300  for i in self.pgrad:
301  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
302  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h, f0 = Ans['X'])
303  self.write_0grads(Ans)
304  self.hct += 1
305  return Ans

601  def printcool_table(self, data=OrderedDict([]), headings=[], banner=None, footnote=None, color=0):
602  """ Print target information in an organized table format. Implemented 6/30 because
603  multiple targets are already printing out tabulated information in very similar ways.
604  This method is a simple wrapper around printcool_dictionary.
605 
606  The input should be something like:
607 
608  @param data Column contents in the form of an OrderedDict, with string keys and list vals.
609  The key is printed in the leftmost column and the vals are printed in the other columns.
610  If non-strings are passed, they will be converted to strings (not recommended).
611 
612  @param headings Column headings in the form of a list. It must be equal to the number to the list length
613  for each of the "vals" in OrderedDict, plus one. Use "\n" characters to specify long
614  column names that may take up more than one line.
615 
616  @param banner Optional heading line, which will be printed at the top in the title.
617  @param footnote Optional footnote line, which will be printed at the bottom.
618 
619  """
620  tline="Target: %s Type: %s Objective = %.5e" % (self.name, self.__class__.__name__, self.objective)
621  nc = len(headings)
622  if banner != None:
623  tlines = [banner, tline]
624  else:
625  tlines = [tline]
626  # Sanity check.
627  for val in data.values():
628  if (len(val)+1) != nc:
629  raise RuntimeError('There are %i column headings, so the values in the data dictionary must be lists of length %i (currently %i)' % (nc, nc-1, len(val)))
630  cwidths = [0 for i in range(nc)]
631  # Figure out maximum column width.
632  # First look at all of the column headings...
633  crows = []
634  for cnum, cname in enumerate(headings):
635  crows.append(len(cname.split('\n')))
636  for l in cname.split('\n'):
637  cwidths[cnum] = max(cwidths[cnum], len(l))
638  # Then look at the row names to stretch out the first column width...
639  for k in data.keys():
640  cwidths[0] = max(cwidths[0], len(str(k)))
641  # Then look at the data values to stretch out the other column widths.
642  for v in data.values():
643  for n, f in enumerate(v):
644  cwidths[n+1] = max(cwidths[n+1], len(str(f)))
645  for i in range(1, len(cwidths)):
646  cwidths[i] += 2
647  if cwidths[0] < 15:
648  cwidths[0] = 15
649  cblocks = [['' for i in range(max(crows) - len(cname.split('\n')))] + cname.split('\n') for cnum, cname in enumerate(headings)]
650  # The formatting line consisting of variable column widths
651  fline = ' '.join("%%%s%is" % (("-" if i==0 else ""), j) for i, j in enumerate(cwidths))
652  vline = ' '.join(["%%%is" % j for i, j in enumerate(cwidths) if i > 0])
653  clines = [fline % (tuple(cblocks[j][i] for j in range(nc))) for i in range(max(crows))]
654  tlines += clines
655  PrintDict = OrderedDict([(key, vline % (tuple(val))) for key, val in data.items()])
656  printcool_dictionary(PrintDict, title='\n'.join(tlines), keywidth=cwidths[0], center=[i==0 for i in range(len(tlines))], leftpad=4, color=color)
