20  """
21  Base class for all fitting targets.
22 
23  In ForceBalance a Target is defined as a set of reference data
24  plus a corresponding method to simulate that data using the force field.
25 
26  The 'computable quantities' may include energies and forces where the
27  reference values come from QM calculations (energy and force matching),
28  energies from an EDA analysis (Maybe in the future, FDA?), molecular
29  properties (like polarizability, refractive indices, multipole moments
30  or vibrational frequencies), relative entropies, and bulk properties.
31  Single-molecule or bulk properties can even come from the experiment!
32 
33  The central idea in ForceBalance is that each quantity makes a
34  contribution to the overall objective function. So we can build force
35  fields that fit several quantities at once, rather than putting all of
36  our chips behind energy and force matching. In the future
37  ForceBalance may even include multiobjective optimization into the
38  optimizer.
39 
40  The optimization is done by way of minimizing an 'objective
41  function', which is comprised of squared differences between the
42  computed and reference values. These differences are not computed
43  in this file, but rather in subclasses that use Target
44  as a base class. Thus, the contents of Target itself
45  are meant to be as general as possible, because the pertinent
46  variables apply to all types of fitting targets.
47 
48  An important node: Target requires that all subclasses
49  have a method get(self,mvals,AGrad=False,AHess=False)
50  that does the following:
51 
52  Inputs:
53  mvals = The parameter vector, which modifies the force field
54  (Note to self: We include mvals with each Target because we can create
55  copies of the force field and do finite difference derivatives)
56  AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
57 
58  Outputs:
59  Answer = {'X': Number, 'G': numpy.array(np), 'H': numpy.array((np,np)) }
60  'X' = The objective function itself
61  'G' = The gradient, elements not computed analytically are zero
62  'H' = The Hessian, elements not computed analytically are zero
63 
64  This is the only global requirement of a Target.
65  Obviously 'get' itself is not defined here, because its
66  calculation will depend entirely on specifically which target
67  we wish to use. However, this should give us a unified framework
68  which will faciliate rapid implementation of Targets.
69 
70  Future work:
71  Robert suggested that I could enable automatic detection of which
72  parameters need to be computed by finite difference. Not a bad idea. :)
73 
74  """

158  def get_G(self,mvals=None):
159  """Computes the objective function contribution and its gradient.
160 
161  First the low-level 'get' method is called with the analytic gradient
162  switch turned on. Then we loop through the fd1_pids and compute
163  the corresponding elements of the gradient by finite difference,
164  if the 'fdgrad' switch is turned on. Alternately we can compute
165  the gradient elements and diagonal Hessian elements at the same time
166  using central difference if 'fdhessdiag' is turned on.
167  """
168  Ans = self.sget(mvals,1,0)
169  for i in range(self.FF.np):
170  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
171  if self.fdhessdiag:
172  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
173  elif self.fdgrad:
174  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
175  self.gct += 1
176  return Ans

178  def get_H(self,mvals=None):
179  """Computes the objective function contribution and its gradient / Hessian.
180 
181  First the low-level 'get' method is called with the analytic gradient
182  and Hessian both turned on. Then we loop through the fd1_pids and compute
183  the corresponding elements of the gradient by finite difference,
184  if the 'fdgrad' switch is turned on.
185 
186  This is followed by looping through the fd2_pids and computing the corresponding
187  Hessian elements by finite difference. Forward finite difference is used
188  throughout for the sake of speed.
189  """
190  Ans = self.sget(mvals,1,1)
191  if self.fdhess:
192  for i in range(self.FF.np):
193  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
194  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
195  for i in range(self.FF.np):
196  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
197  FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
198  Ans['H'][i,:] = FDSlice
199  Ans['H'][:,i] = FDSlice
200  elif self.fdhessdiag:
201  for i in range(self.FF.np):
202  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
203  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h, f0 = Ans['X'])
204  self.hct += 1
205  return Ans

337  def printcool_table(self, data=OrderedDict([]), headings=[], banner=None, footnote=None, color=0):
338  """ Print target information in an organized table format. Implemented 6/30 because
339  multiple targets are already printing out tabulated information in very similar ways.
340  This method is a simple wrapper around printcool_dictionary.
341 
342  The input should be something like:
343 
344  @param data Column contents in the form of an OrderedDict, with string keys and list vals.
345  The key is printed in the leftmost column and the vals are printed in the other columns.
346  If non-strings are passed, they will be converted to strings (not recommended).
347 
348  @param headings Column headings in the form of a list. It must be equal to the number to the list length
349  for each of the "vals" in OrderedDict, plus one. Use "\n" characters to specify long
350  column names that may take up more than one line.
351 
352  @param banner Optional heading line, which will be printed at the top in the title.
353  @param footnote Optional footnote line, which will be printed at the bottom.
354 
355  """
356  tline="Target: %s Type: %s Objective = %.5e" % (self.name, self.__class__.__name__, self.objective)
357  nc = len(headings)
358  if banner != None:
359  tlines = [banner, tline]
360  else:
361  tlines = [tline]
362  # Sanity check.
363  for val in data.values():
364  if (len(val)+1) != nc:
365  raise RuntimeError('There are %i column headings, so the values in the data dictionary must be lists of length %i (currently %i)' % (nc, nc-1, len(val)))
366  cwidths = [0 for i in range(nc)]
367  # Figure out maximum column width.
368  # First look at all of the column headings...
369  crows = []
370  for cnum, cname in enumerate(headings):
371  crows.append(len(cname.split('\n')))
372  for l in cname.split('\n'):
373  cwidths[cnum] = max(cwidths[cnum], len(l))
374  # Then look at the row names to stretch out the first column width...
375  for k in data.keys():
376  cwidths[0] = max(cwidths[0], len(k))
377  # Then look at the data values to stretch out the other column widths.
378  for v in data.values():
379  for n, f in enumerate(v):
380  cwidths[n+1] = max(cwidths[n+1], len(str(f)))
381  for i in range(1, len(cwidths)):
382  cwidths[i] += 2
383  if cwidths[0] < 25:
384  cwidths[0] = 25
385  cblocks = [['' for i in range(max(crows) - len(cname.split('\n')))] + cname.split('\n') for cnum, cname in enumerate(headings)]
386  # The formatting line consisting of variable column widths
387  fline = ' '.join("%%%s%is" % (("-" if i==0 else ""), j) for i, j in enumerate(cwidths))
388  vline = ' '.join(["%%%is" % j for i, j in enumerate(cwidths) if i > 0])
389  clines = [fline % (tuple(cblocks[j][i] for j in range(nc))) for i in range(max(crows))]
390  tlines += clines
391  PrintDict = OrderedDict([(key, vline % (tuple(val))) for key, val in data.items()])
392  printcool_dictionary(PrintDict, title='\n'.join(tlines), keywidth=cwidths[0], center=False, leftpad=4, color=color)
