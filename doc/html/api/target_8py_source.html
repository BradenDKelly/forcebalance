00017     """
00018     Base class for all fitting targets.
00019     
00020     In ForceBalance a Target is defined as a set of reference data
00021     plus a corresponding method to simulate that data using the force field.
00022     
00023     The 'computable quantities' may include energies and forces where the
00024     reference values come from QM calculations (energy and force matching),
00025     energies from an EDA analysis (Maybe in the future, FDA?), molecular
00026     properties (like polarizability, refractive indices, multipole moments
00027     or vibrational frequencies), relative entropies, and bulk properties.
00028     Single-molecule or bulk properties can even come from the experiment!
00029     
00030     The central idea in ForceBalance is that each quantity makes a
00031     contribution to the overall objective function.  So we can build force
00032     fields that fit several quantities at once, rather than putting all of
00033     our chips behind energy and force matching.  In the future
00034     ForceBalance may even include multiobjective optimization into the
00035     optimizer.
00036     
00037     The optimization is done by way of minimizing an 'objective
00038     function', which is comprised of squared differences between the
00039     computed and reference values.  These differences are not computed
00040     in this file, but rather in subclasses that use Target
00041     as a base class.  Thus, the contents of Target itself
00042     are meant to be as general as possible, because the pertinent
00043     variables apply to all types of fitting targets.
00044 
00045     An important node: Target requires that all subclasses
00046     have a method get(self,mvals,AGrad=False,AHess=False)
00047     that does the following:
00048     
00049     Inputs: 
00050     mvals        = The parameter vector, which modifies the force field
00051     (Note to self: We include mvals with each Target because we can create
00052     copies of the force field and do finite difference derivatives)
00053     AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
00054 
00055     Outputs:
00056     Answer       = {'X': Number, 'G': numpy.array(np), 'H': numpy.array((np,np)) }
00057     'X'          = The objective function itself
00058     'G'          = The gradient, elements not computed analytically are zero
00059     'H'          = The Hessian, elements not computed analytically are zero
00060 
00061     This is the only global requirement of a Target.
00062     Obviously 'get' itself is not defined here, because its
00063     calculation will depend entirely on specifically which target
00064     we wish to use.  However, this should give us a unified framework
00065     which will faciliate rapid implementation of Targets.
00066 
00067     Future work:
00068     Robert suggested that I could enable automatic detection of which
00069     parameters need to be computed by finite difference.  Not a bad idea. :)
00070 
00071     """

00155     def get_G(self,mvals=None):
00156         """Computes the objective function contribution and its gradient.
00157 
00158         First the low-level 'get' method is called with the analytic gradient
00159         switch turned on.  Then we loop through the fd1_pids and compute
00160         the corresponding elements of the gradient by finite difference,
00161         if the 'fdgrad' switch is turned on.  Alternately we can compute
00162         the gradient elements and diagonal Hessian elements at the same time
00163         using central difference if 'fdhessdiag' is turned on.
00164         """
00165         Ans = self.sget(mvals,1,0)
00166         for i in range(self.FF.np):
00167             if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
00168                 if self.fdhessdiag:
00169                     Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00170                 elif self.fdgrad:
00171                     Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00172         self.gct += 1
00173         return Ans

00175     def get_H(self,mvals=None):
00176         """Computes the objective function contribution and its gradient / Hessian.
00177 
00178         First the low-level 'get' method is called with the analytic gradient
00179         and Hessian both turned on.  Then we loop through the fd1_pids and compute
00180         the corresponding elements of the gradient by finite difference,
00181         if the 'fdgrad' switch is turned on.
00182 
00183         This is followed by looping through the fd2_pids and computing the corresponding
00184         Hessian elements by finite difference.  Forward finite difference is used
00185         throughout for the sake of speed.
00186         """
00187         Ans = self.sget(mvals,1,1)
00188         if self.fdhess:
00189             for i in range(self.FF.np):
00190                 if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
00191                     Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00192             for i in range(self.FF.np):
00193                 if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
00194                     FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
00195                     Ans['H'][i,:] = FDSlice

00337         This method is a simple wrapper around printcool_dictionary.  
00338 
00339         The input should be something like:
00340 
00341         @param data Column contents in the form of an OrderedDict, with string keys and list vals.
00342         The key is printed in the leftmost column and the vals are printed in the other columns.
00343         If non-strings are passed, they will be converted to strings (not recommended).
00344         
00345         @param headings Column headings in the form of a list.  It must be equal to the number to the list length

00349         @param banner Optional heading line, which will be printed at the top in the title.
00350         @param footnote Optional footnote line, which will be printed at the bottom.
00351         
00352         """
00353         tline="Target: %s Type: %s Objective = %.5e" % (self.name, self.__class__.__name__, self.objective)
00354         nc = len(headings)
00355         if banner != None:
00356             tlines = [banner, tline]
00357         else:
00358             tlines = [tline]
00359         # Sanity check.
00360         for val in data.values():
00361             if (len(val)+1) != nc:
00362                 raise RuntimeError('There are %i column headings, so the values in the data dictionary must be lists of length %i (currently %i)' % (nc, nc-1, len(val)))
00363         cwidths = [0 for i in range(nc)]
00364         # Figure out maximum column width.
00365         # First look at all of the column headings...
00366         crows = []
00367         for cnum, cname in enumerate(headings):
00368             crows.append(len(cname.split('\n')))
00369             for l in cname.split('\n'):
00370                 cwidths[cnum] = max(cwidths[cnum], len(l))
00371         # Then look at the row names to stretch out the first column width...
00372         for k in data.keys():
00373             cwidths[0] = max(cwidths[0], len(k))
00374         # Then look at the data values to stretch out the other column widths.
00375         for v in data.values():
00376             for n, f in enumerate(v):
00377                 cwidths[n+1] = max(cwidths[n+1], len(str(f)))
00378         for i in range(1, len(cwidths)):
00379             cwidths[i] += 2
00380         if cwidths[0] < 25:
00381             cwidths[0] = 25
00382         cblocks = [['' for i in range(max(crows) - len(cname.split('\n')))] + cname.split('\n') for cnum, cname in enumerate(headings)]
