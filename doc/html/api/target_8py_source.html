17  """
18  Base class for all fitting targets.
19 
20  In ForceBalance a Target is defined as a set of reference data
21  plus a corresponding method to simulate that data using the force field.
22 
23  The 'computable quantities' may include energies and forces where the
24  reference values come from QM calculations (energy and force matching),
25  energies from an EDA analysis (Maybe in the future, FDA?), molecular
26  properties (like polarizability, refractive indices, multipole moments
27  or vibrational frequencies), relative entropies, and bulk properties.
28  Single-molecule or bulk properties can even come from the experiment!
29 
30  The central idea in ForceBalance is that each quantity makes a
31  contribution to the overall objective function. So we can build force
32  fields that fit several quantities at once, rather than putting all of
33  our chips behind energy and force matching. In the future
34  ForceBalance may even include multiobjective optimization into the
35  optimizer.
36 
37  The optimization is done by way of minimizing an 'objective
38  function', which is comprised of squared differences between the
39  computed and reference values. These differences are not computed
40  in this file, but rather in subclasses that use Target
41  as a base class. Thus, the contents of Target itself
42  are meant to be as general as possible, because the pertinent
43  variables apply to all types of fitting targets.
44 
45  An important node: Target requires that all subclasses
46  have a method get(self,mvals,AGrad=False,AHess=False)
47  that does the following:
48 
49  Inputs:
50  mvals = The parameter vector, which modifies the force field
51  (Note to self: We include mvals with each Target because we can create
52  copies of the force field and do finite difference derivatives)
53  AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
54 
55  Outputs:
56  Answer = {'X': Number, 'G': numpy.array(np), 'H': numpy.array((np,np)) }
57  'X' = The objective function itself
58  'G' = The gradient, elements not computed analytically are zero
59  'H' = The Hessian, elements not computed analytically are zero
60 
61  This is the only global requirement of a Target.
62  Obviously 'get' itself is not defined here, because its
63  calculation will depend entirely on specifically which target
64  we wish to use. However, this should give us a unified framework
65  which will faciliate rapid implementation of Targets.
66 
67  Future work:
68  Robert suggested that I could enable automatic detection of which
69  parameters need to be computed by finite difference. Not a bad idea. :)
70 
71  """

155  def get_G(self,mvals=None):
156  """Computes the objective function contribution and its gradient.
157 
158  First the low-level 'get' method is called with the analytic gradient
159  switch turned on. Then we loop through the fd1_pids and compute
160  the corresponding elements of the gradient by finite difference,
161  if the 'fdgrad' switch is turned on. Alternately we can compute
162  the gradient elements and diagonal Hessian elements at the same time
163  using central difference if 'fdhessdiag' is turned on.
164  """
165  Ans = self.sget(mvals,1,0)
166  for i in range(self.FF.np):
167  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
168  if self.fdhessdiag:
169  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
170  elif self.fdgrad:
171  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
172  self.gct += 1
173  return Ans

175  def get_H(self,mvals=None):
176  """Computes the objective function contribution and its gradient / Hessian.
177 
178  First the low-level 'get' method is called with the analytic gradient
179  and Hessian both turned on. Then we loop through the fd1_pids and compute
180  the corresponding elements of the gradient by finite difference,
181  if the 'fdgrad' switch is turned on.
182 
183  This is followed by looping through the fd2_pids and computing the corresponding
184  Hessian elements by finite difference. Forward finite difference is used
185  throughout for the sake of speed.
186  """
187  Ans = self.sget(mvals,1,1)
188  if self.fdhess:
189  for i in range(self.FF.np):
190  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
191  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
192  for i in range(self.FF.np):
193  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
194  FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
195  Ans['H'][i,:] = FDSlice
196  Ans['H'][:,i] = FDSlice
197  elif self.fdhessdiag:
198  for i in range(self.FF.np):
199  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
200  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h, f0 = Ans['X'])
201  self.hct += 1
202  return Ans

334  def printcool_table(self, data=OrderedDict([]), headings=[], banner=None, footnote=None, color=0):
335  """ Print target information in an organized table format. Implemented 6/30 because
336  multiple targets are already printing out tabulated information in very similar ways.
337  This method is a simple wrapper around printcool_dictionary.
338 
339  The input should be something like:
340 
341  @param data Column contents in the form of an OrderedDict, with string keys and list vals.
342  The key is printed in the leftmost column and the vals are printed in the other columns.
343  If non-strings are passed, they will be converted to strings (not recommended).
344 
345  @param headings Column headings in the form of a list. It must be equal to the number to the list length
346  for each of the "vals" in OrderedDict, plus one. Use "\n" characters to specify long
347  column names that may take up more than one line.
348 
349  @param banner Optional heading line, which will be printed at the top in the title.
350  @param footnote Optional footnote line, which will be printed at the bottom.
351 
352  """
353  tline="Target: %s Type: %s Objective = %.5e" % (self.name, self.__class__.__name__, self.objective)
354  nc = len(headings)
355  if banner != None:
356  tlines = [banner, tline]
357  else:
358  tlines = [tline]
359  # Sanity check.
360  for val in data.values():
361  if (len(val)+1) != nc:
362  raise RuntimeError('There are %i column headings, so the values in the data dictionary must be lists of length %i (currently %i)' % (nc, nc-1, len(val)))
363  cwidths = [0 for i in range(nc)]
364  # Figure out maximum column width.
365  # First look at all of the column headings...
366  crows = []
367  for cnum, cname in enumerate(headings):
368  crows.append(len(cname.split('\n')))
369  for l in cname.split('\n'):
370  cwidths[cnum] = max(cwidths[cnum], len(l))
371  # Then look at the row names to stretch out the first column width...
372  for k in data.keys():
373  cwidths[0] = max(cwidths[0], len(k))
374  # Then look at the data values to stretch out the other column widths.
375  for v in data.values():
376  for n, f in enumerate(v):
377  cwidths[n+1] = max(cwidths[n+1], len(str(f)))
378  for i in range(1, len(cwidths)):
379  cwidths[i] += 2
380  if cwidths[0] < 25:
381  cwidths[0] = 25
382  cblocks = [['' for i in range(max(crows) - len(cname.split('\n')))] + cname.split('\n') for cnum, cname in enumerate(headings)]
383  # The formatting line consisting of variable column widths
384  fline = ' '.join("%%%s%is" % (("-" if i==0 else ""), j) for i, j in enumerate(cwidths))
385  vline = ' '.join(["%%%is" % j for i, j in enumerate(cwidths) if i > 0])
386  clines = [fline % (tuple(cblocks[j][i] for j in range(nc))) for i in range(max(crows))]
387  tlines += clines
388  PrintDict = OrderedDict([(key, vline % (tuple(val))) for key, val in data.items()])
389  printcool_dictionary(PrintDict, title='\n'.join(tlines), keywidth=cwidths[0], center=False, leftpad=4, color=color)
