00020     """
00021     Base class for all fitting targets.
00022     
00023     In ForceBalance a Target is defined as a set of reference data
00024     plus a corresponding method to simulate that data using the force field.
00025     
00026     The 'computable quantities' may include energies and forces where the
00027     reference values come from QM calculations (energy and force matching),
00028     energies from an EDA analysis (Maybe in the future, FDA?), molecular
00029     properties (like polarizability, refractive indices, multipole moments
00030     or vibrational frequencies), relative entropies, and bulk properties.
00031     Single-molecule or bulk properties can even come from the experiment!
00032     
00033     The central idea in ForceBalance is that each quantity makes a
00034     contribution to the overall objective function.  So we can build force
00035     fields that fit several quantities at once, rather than putting all of
00036     our chips behind energy and force matching.  In the future
00037     ForceBalance may even include multiobjective optimization into the
00038     optimizer.
00039     
00040     The optimization is done by way of minimizing an 'objective
00041     function', which is comprised of squared differences between the
00042     computed and reference values.  These differences are not computed
00043     in this file, but rather in subclasses that use Target
00044     as a base class.  Thus, the contents of Target itself
00045     are meant to be as general as possible, because the pertinent
00046     variables apply to all types of fitting targets.
00047 
00048     An important node: Target requires that all subclasses
00049     have a method get(self,mvals,AGrad=False,AHess=False)
00050     that does the following:
00051     
00052     Inputs: 
00053     mvals        = The parameter vector, which modifies the force field
00054     (Note to self: We include mvals with each Target because we can create
00055     copies of the force field and do finite difference derivatives)
00056     AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
00057 
00058     Outputs:
00059     Answer       = {'X': Number, 'G': numpy.array(np), 'H': numpy.array((np,np)) }
00060     'X'          = The objective function itself
00061     'G'          = The gradient, elements not computed analytically are zero
00062     'H'          = The Hessian, elements not computed analytically are zero
00063 
00064     This is the only global requirement of a Target.
00065     Obviously 'get' itself is not defined here, because its
00066     calculation will depend entirely on specifically which target
00067     we wish to use.  However, this should give us a unified framework
00068     which will faciliate rapid implementation of Targets.
00069 
00070     Future work:
00071     Robert suggested that I could enable automatic detection of which
00072     parameters need to be computed by finite difference.  Not a bad idea. :)
00073 
00074     """

00158     def get_G(self,mvals=None):
00159         """Computes the objective function contribution and its gradient.
00160 
00161         First the low-level 'get' method is called with the analytic gradient
00162         switch turned on.  Then we loop through the fd1_pids and compute
00163         the corresponding elements of the gradient by finite difference,
00164         if the 'fdgrad' switch is turned on.  Alternately we can compute
00165         the gradient elements and diagonal Hessian elements at the same time
00166         using central difference if 'fdhessdiag' is turned on.
00167         """
00168         Ans = self.sget(mvals,1,0)
00169         for i in range(self.FF.np):
00170             if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
00171                 if self.fdhessdiag:
00172                     Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00173                 elif self.fdgrad:
00174                     Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00175         self.gct += 1
00176         return Ans

00178     def get_H(self,mvals=None):
00179         """Computes the objective function contribution and its gradient / Hessian.
00180 
00181         First the low-level 'get' method is called with the analytic gradient
00182         and Hessian both turned on.  Then we loop through the fd1_pids and compute
00183         the corresponding elements of the gradient by finite difference,
00184         if the 'fdgrad' switch is turned on.
00185 
00186         This is followed by looping through the fd2_pids and computing the corresponding
00187         Hessian elements by finite difference.  Forward finite difference is used
00188         throughout for the sake of speed.
00189         """
00190         Ans = self.sget(mvals,1,1)
00191         if self.fdhess:
00192             for i in range(self.FF.np):
00193                 if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
00194                     Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
00195             for i in range(self.FF.np):
00196                 if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
00197                     FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
00198                     Ans['H'][i,:] = FDSlice
00199                     Ans['H'][:,i] = FDSlice
00200         elif self.fdhessdiag:
00201             for i in range(self.FF.np):
00202                 if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
00203                     Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h, f0 = Ans['X'])

00337     def printcool_table(self, data=OrderedDict([]), headings=[], banner=None, footnote=None, color=0):
00338         """ Print target information in an organized table format.  Implemented 6/30 because
00339         multiple targets are already printing out tabulated information in very similar ways.
00340         This method is a simple wrapper around printcool_dictionary.  
00341 
00342         The input should be something like:
00343 
00344         @param data Column contents in the form of an OrderedDict, with string keys and list vals.
00345         The key is printed in the leftmost column and the vals are printed in the other columns.
00346         If non-strings are passed, they will be converted to strings (not recommended).
00347         
00348         @param headings Column headings in the form of a list.  It must be equal to the number to the list length
00349         for each of the "vals" in OrderedDict, plus one.  Use "\n" characters to specify long
00350         column names that may take up more than one line.
00351 
00352         @param banner Optional heading line, which will be printed at the top in the title.
00353         @param footnote Optional footnote line, which will be printed at the bottom.
00354         
00355         """
00356         tline="Target: %s Type: %s Objective = %.5e" % (self.name, self.__class__.__name__, self.objective)
00357         nc = len(headings)
00358         if banner != None:
00359             tlines = [banner, tline]
00360         else:
00361             tlines = [tline]
00362         # Sanity check.
00363         for val in data.values():
00364             if (len(val)+1) != nc:
00365                 raise RuntimeError('There are %i column headings, so the values in the data dictionary must be lists of length %i (currently %i)' % (nc, nc-1, len(val)))
00366         cwidths = [0 for i in range(nc)]
00367         # Figure out maximum column width.
00368         # First look at all of the column headings...
00369         crows = []
00370         for cnum, cname in enumerate(headings):
00371             crows.append(len(cname.split('\n')))
00372             for l in cname.split('\n'):
00373                 cwidths[cnum] = max(cwidths[cnum], len(l))
00374         # Then look at the row names to stretch out the first column width...
00375         for k in data.keys():
00376             cwidths[0] = max(cwidths[0], len(k))
00377         # Then look at the data values to stretch out the other column widths.
00378         for v in data.values():
00379             for n, f in enumerate(v):
00380                 cwidths[n+1] = max(cwidths[n+1], len(str(f)))
00381         for i in range(1, len(cwidths)):
00382             cwidths[i] += 2
00383         if cwidths[0] < 25:
00384             cwidths[0] = 25
00385         cblocks = [['' for i in range(max(crows) - len(cname.split('\n')))] + cname.split('\n') for cnum, cname in enumerate(headings)]
00386         # The formatting line consisting of variable column widths
00387         fline = ' '.join("%%%s%is" % (("-" if i==0 else ""), j) for i, j in enumerate(cwidths))
00388         vline = ' '.join(["%%%is" % j for i, j in enumerate(cwidths) if i > 0])
00389         clines = [fline % (tuple(cblocks[j][i] for j in range(nc))) for i in range(max(crows))]
