198  def compute_netforce_torque(self, xyz, force, QM=False):
199  # Convert an array of (3 * n_atoms) atomistic forces
200  # to an array of (3 * (n_forces + n_torques)) net forces and torques.
201  # This code is rather slow. It requires the system to have a list
202  # of masses and blocking numbers.
203  if not self.topology_flag:
204  raise Exception('Cannot do net forces and torques for class %s because read_topology is not implemented' % self.__class__.__name__)
205 
206  if self.force_map == 'molecule':
207  Block = self.AtomLists['MoleculeNumber']
208  elif self.force_map == 'residue':
209  Block = self.AtomLists['ResidueNumber']
210  elif self.force_map == 'chargegroup':
211  Block = self.AtomLists['ChargeGroupNumber']
212  else:
213  raise Exception('Please choose a valid force_map keyword: molecule, residue, chargegroup')
214 
215  # Try to be intelligent here. Before computing net forces and torques, first filter out all particles that are not atoms.
216  if len(xyz) > self.fitatoms:
217  xyz1 = array([xyz[i] for i in range(len(xyz)) if self.AtomLists['ParticleType'][i] == 'A'])
218  else:
219  xyz1 = xyz.copy()
220 
221  if len(Block) > self.fitatoms:
222  Block = [Block[i] for i in range(len(Block)) if self.AtomLists['ParticleType'][i] == 'A']
223 
224  if len(self.AtomLists['Mass']) > self.fitatoms:
225  Mass = array([self.AtomLists['Mass'][i] for i in range(len(xyz)) if self.AtomLists['ParticleType'][i] == 'A'])
226  else:
227  Mass = array(self.AtomLists['Mass'])
228 
229  NetForces = []
230  Torques = []
231  for b in sorted(set(Block)):
232  AtomBlock = array([i for i in range(len(Block)) if Block[i] == b])
233  CrdBlock = array(list(itertools.chain(*[range(3*i, 3*i+3) for i in AtomBlock])))
234  com = sum(xyz1[AtomBlock]*outer(Mass[AtomBlock],ones(3)), axis=0) / sum(Mass[AtomBlock])
235  frc = force[CrdBlock].reshape(-1,3)
236  NetForce = sum(frc, axis=0)
237  xyzb = xyz1[AtomBlock]
238  Torque = zeros(3, dtype=float)
239  for a in range(len(xyzb)):
240  R = xyzb[a] - com
241  F = frc[a]
242  # I think the unit of torque is in nm x kJ / nm.
243  Torque += cross(R, F) / 10
244  NetForces += [i for i in NetForce]
245  # Increment the torques only if we have more than one atom in our block.
246  if len(xyzb) > 1:
247  Torques += [i for i in Torque]
248  netfrc_torque = array(NetForces + Torques)
249  self.nnf = len(NetForces)/3

255  """ Read the reference ab initio data from a file such as qdata.txt.
256 
257  @todo Add an option for picking any slice out of
258  qdata.txt, helpful for cross-validation
259 
260  @todo Closer integration of reference data with program -
261  leave behind the qdata.txt format? (For now, I like the
262  readability of qdata.txt)
263 
264  After reading in the information from qdata.txt, it is converted
265  into the GROMACS energy units (kind of an arbitrary choice);
266  forces (kind of a misnomer in qdata.txt) are multipled by -1
267  to convert gradients to forces.
268 
269  We also subtract out the mean energies of all energy arrays
270  because energy/force matching does not account for zero-point
271  energy differences between MM and QM (i.e. energy of electrons
272  in core orbitals).
273 
274  The configurations in force/energy matching typically come
275  from a the thermodynamic ensemble of the MM force field at
276  some temperature (by running MD, for example), and for many
277  reasons it is helpful to introduce non-Boltzmann weights in
278  front of these configurations. There are two options: WHAM
279  Boltzmann weights (for combining the weights of several
280  simulations together) and QM Boltzmann weights (for converting
281  MM weights into QM weights). Note that the two sets of weights
282  'stack'; i.e. they can be used at the same time.
283 
284  A 'hybrid' ensemble is possible where we use 50% MM and 50% QM
285  weights. Please read more in LPW and Troy Van Voorhis, JCP
286  Vol. 133, Pg. 231101 (2010), doi:10.1063/1.3519043.
287 
288  @todo The WHAM Boltzmann weights are generated by external
289  scripts (wanalyze.py and make-wham-data.sh) and passed in;
290  perhaps these scripts can be added to the ForceBalance
291  distribution or integrated more tightly.
292 
293  Finally, note that using non-Boltzmann weights degrades the
294  statistical information content of the snapshots. This
295  problem will generally become worse if the ensemble to which
296  we're reweighting is dramatically different from the one we're
297  sampling from. We end up with a set of Boltzmann weights like
298  [1e-9, 1e-9, 1.0, 1e-9, 1e-9 ... ] and this is essentially just
299  one snapshot. I believe Troy is working on something to cure
300  this problem.
301 
302  Here, we have a measure for the information content of our snapshots,
303  which comes easily from the definition of information entropy:
304 
305  S = -1*Sum_i(P_i*log(P_i))
306  InfoContent = exp(-S)
307 
308  With uniform weights, InfoContent is equal to the number of snapshots;
309  with horrible weights, InfoContent is closer to one.

462  def get_energy_force_(self, mvals, AGrad=False, AHess=False):
463  """
464  LPW 06-30-2013
465 
466  This subroutine builds the objective function (and optionally
467  its derivatives) from a general simulation software. This is
468  in contrast to using GROMACS-X2, which computes the objective
469  function and prints it out; then 'get' only needs to call
470  GROMACS and read it in.
471 
472  This subroutine interfaces with simulation software 'drivers'.
473  The driver is only expected to give the energy and forces.
474 
475  Now this subroutine may sound trivial since the objective
476  function is simply a least-squares quantity (M-Q)^2 - but
477  there are a number of nontrivial considerations. I will list
478  them here.
479 
480  0) Polytensor formulation: Because there may exist covariance
481  between different components of the force (or covariance
482  between the energy and the force), we build the objective
483  function by taking outer products of vectors that have the
484  form [E F_1x F_1y F_1z F_2x F_2y ... ], and then we trace it
485  with the inverse of the covariance matrix to get the objective
486  function.
487 
488  This version implements both the polytensor formulation and the standard formulation.
489 
490  1) Boltzmann weights and normalization: Each snapshot has its
491  own Boltzmann weight, which may or may not be normalized.
492  This subroutine does the normalization automatically.
493 
494  2) Subtracting out the mean energy gap: The zero-point energy
495  difference between reference data and simulation is
496  meaningless. This subroutine subtracts it out.
497 
498  3) Hybrid ensembles: This program builds a combined objective
499  function from both MM and QM ensembles, which is rigorously
500  better than using a single ensemble.
501 
502  Note that this subroutine does not do EVERYTHING that
503  GROMACS-X2 can do, which includes:

712  if not cv:
713  for p in range(np):
714  if not AGrad: continue
715  if self.all_at_once:
716  M_p[p] = dM_all[i, p]
717  M_pp[p] = ddM_all[i, p]
718  else:

753  Q_all_print = hstack((col(self.eqm),self.fref))
754  else:
755  Q_all_print = col(self.eqm)
756  if not self.absolute:
757  Q_all_print[:,0] -= mean(Q_all_print[:,0])

760  savetxt('Q.txt',Q_all_print)

762  if self.writelevel > 0:
763  savetxt('QM-vs-MM-energies.txt',EnergyComparison)
764  if self.force:
765  # Write .xyz files which can be viewed in vmd.
766  try:
767  # Only print forces on true atoms, and ignore virtual sites.
768  TrueAtoms = array([i for i in range(self.traj.na) if self.AtomLists['ParticleType'][i] == 'A'])
769  except:
770  TrueAtoms = arange(self.traj.na)
771  QMTraj = self.traj[:].atom_select(TrueAtoms)
772  Mforce_obj = QMTraj[:]
773  Qforce_obj = QMTraj[:]
774  Mforce_print = array(M_all_print[:,1:3*self.fitatoms+1])
775  Qforce_print = array(Q_all_print[:,1:3*self.fitatoms+1])
776  Dforce_norm = array([linalg.norm(Mforce_print[i,:] - Qforce_print[i,:]) for i in range(ns)])

780  for i in range(ns):

798  savetxt('Dforce_norm.dat', Dforce_norm)

802  # STEP 3: Build the (co)variance matrix and invert it. #

806  print "Done with snapshots, building objective function now\r",

829  block = 3*self.nnf
830  end = start + block

835  end = start + block

905  # Get the other half of the Hessian matrix.
906  X2_M_pq[q,p] = X2_M_pq[p,q]
907  X2_Q_pq[q,p] = X2_Q_pq[p,q]
908  #==============================================================#
909  # STEP 5: Build the return quantities. #
910  #==============================================================#
911  # The objective function
912  X2 = MBP * X2_M + QBP * X2_Q
913  if not cv:
914  # Derivatives of the objective function
915  G = zeros(np,dtype=float)
916  H = zeros((np,np),dtype=float)
917  for p in range(np):
918  if not AGrad: continue
919  G[p] = MBP * X2_M_p[p] + QBP * X2_Q_p[p]
920  if not AHess: continue
921  for q in range(np):
922  H[p,q] = MBP * X2_M_pq[p,q] + QBP * X2_Q_pq[p,q]
923  # Energy error in kJ/mol
924  if not self.absolute:
925  E0_M = (2*Q0_M[0]*M0_M[0] - Q0_M[0]*Q0_M[0] - M0_M[0]*M0_M[0])/Z/Z;
926  E0_Q = (2*Q0_Q[0]*M0_Q[0] - Q0_Q[0]*Q0_Q[0] - M0_Q[0]*M0_Q[0])/Y/Y;
927  else:
928  E0_M = 0.0
929  E0_Q = 0.0
930  if cv:
931  dE = MBP * sqrt(SPiXi[0,0]/Z + E0_M) + QBP * sqrt(SRiXi[0,0]/Y + E0_Q)
932  else:
933  dE = MBP * sqrt(SPiXi[0]/Z + E0_M) + QBP * sqrt(SRiXi[0]/Y + E0_Q)
934  # Fractional energy error.
935  dEfrac = MBP * sqrt((SPiXi[0]/Z + E0_M) / (QQ_M[0]/Z - Q0_M[0]**2/Z/Z)) + QBP * sqrt((SRiXi[0]/Y + E0_Q) / (QQ_Q[0]/Y - Q0_Q[0]**2/Y/Y))
936  # Absolute and Fractional force error.
937  if self.force:
938  dF = MBP * dF_M / Z + QBP * dF_Q / Y
939  qF = MBP * qF_M / Z + QBP * qF_Q / Y
940  dFfrac = MBP * (dF_M/qF_M) + QBP * (dF_Q/qF_Q)
941  # Old code for generating percentage force errors is less intuitive.
