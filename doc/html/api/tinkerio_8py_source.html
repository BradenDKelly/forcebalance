00059 
00060     This class is instantiated when we begin to read in a file.  The
00061     feed(line) method updates the state of the machine, informing it

00063     look up the interaction type and parameter type for building the
00064     parameter ID.

00075     def feed(self, line):
00076         """ Given a line, determine the interaction type and the atoms involved (the suffix).
00077 
00078         TINKER generally has stuff like this:
00079 
00080         @verbatim
00081         bond-cubic              -2.55
00082         bond-quartic            3.793125
00083 
00084         vdw           1               3.4050     0.1100
00085         vdw           2               2.6550     0.0135      0.910 # PARM 4
00086 
00087         multipole     2    1    2               0.25983
00088                                                -0.03859    0.00000   -0.05818
00089                                                -0.03673
00090                                                 0.00000   -0.10739
00091                                                -0.00203    0.00000    0.14412
00092         @endverbatim
00093 
00094         The '#PARM 4' has no effect on TINKER but it indicates that we
00095         are tuning the fourth field on the line (the 0.910 value).
00096 
00097         @todo Put the rescaling factors for TINKER parameters in here.
00098         Currently we're using the initial value to determine the
00099         rescaling factor which is not very good.
00100 
00101         Every parameter line is prefaced by the interaction type
00102         except for 'multipole' which is on multiple lines.  Because
00103         the lines that come after 'multipole' are predictable, we just
00104         determine the current line using the previous line.
00105 
00106         Random note: Unit of force is kcal / mole / angstrom squared.
00107         
00108         """
00109         s          = line.split()
00110         self.ln   += 1
00111         # No sense in doing anything for an empty line or a comment line.
00112         if len(s) == 0 or match('^#',line): return None, None
00113         # From the line, figure out the interaction type.  If this line doesn't correspond
00114         # to an interaction, then do nothing.
00115         if s[0].upper() in pdict:
00116             self.itype = s[0].upper()
00117         # This is kind of a silly hack that allows us to take care of the 'multipole' keyword,
00118         # because of the syntax that the TINKER .prm file uses.
00119         elif s[0].upper() == 'MULTIPOLE':
00120             self.itype = 'MCHARGE'
00121         elif self.itype == 'MCHARGE':
00122             self.itype = 'DIPOLE'
00123         elif self.itype == 'DIPOLE':
00124             self.itype = 'QUADX'
00125         elif self.itype == 'QUADX':
00126             self.itype = 'QUADY'
00127         elif self.itype == 'QUADY':
00128             self.itype = 'QUADZ'
00129         else:
00130             self.itype = None
00131 
00132         if self.itype in pdict:
00133             if 'Atom' in pdict[self.itype]:
00134                 # List the atoms in the interaction.
00135                 self.atom = [s[i] for i in pdict[self.itype]['Atom']]
00136             # The suffix of the parameter ID is built from the atom    #
00137             # types/classes involved in the interaction.
00138             self.suffix = '.'.join(self.atom)

00179 def modify_key(src, in_dict):
00180     """ Performs in-place modification of a TINKER .key file. 
00181 
00182     The input dictionary contains key:value pairs such as
00183     "polarization direct".  If the key exists in the TINKER file, then

00197     """

00200         fin = open(src).readlines()
00201     else:
00202         raise Exception("This function shouldn't be used to follow symbolic links, because I don't want to modify files in the target directory")
00203     odict = OrderedDict([(key.lower(), val) for key, val in in_dict.items()])
00204     flags = OrderedDict([(key, False) for key in odict.keys()])
00205     outlines = []
00206     for line in open(src).readlines():
00207         s = line.split()
00208         if len(s) == 0:
00209             outlines.append(line)
00210             continue

00216                 outlines.append("%s %s\n" % (key, val))
00217             else:
00218                 outlines.append("%s\n" % (key))
00219             flags[key] = True
00220         else:
00221             outlines.append(line)
00222     for key, val in odict.items():
00223         if not flags[key]:
00224             if val != None:
00225                 outlines.append("%s %s\n" % (key, val))
