00072     def feed(self, line):
00073         """ Given a line, determine the interaction type and the atoms involved (the suffix).
00074 
00075         TINKER generally has stuff like this:
00076 
00077         @verbatim
00078         bond-cubic              -2.55
00079         bond-quartic            3.793125
00080 
00081         vdw           1               3.4050     0.1100
00082         vdw           2               2.6550     0.0135      0.910 # PARM 4
00083 
00084         multipole     2    1    2               0.25983
00085                                                -0.03859    0.00000   -0.05818
00086                                                -0.03673
00087                                                 0.00000   -0.10739
00088                                                -0.00203    0.00000    0.14412
00089         @endverbatim
00090 
00091         The '#PARM 4' has no effect on TINKER but it indicates that we
00092         are tuning the fourth field on the line (the 0.910 value).
00093 
00094         @todo Put the rescaling factors for TINKER parameters in here.
00095         Currently we're using the initial value to determine the
00096         rescaling factor which is not very good.
00097 
00098         Every parameter line is prefaced by the interaction type
00099         except for 'multipole' which is on multiple lines.  Because
00100         the lines that come after 'multipole' are predictable, we just
00101         determine the current line using the previous line.
00102 
00103         Random note: Unit of force is kcal / mole / angstrom squared.
00104         
00105         """
00106         s          = line.split()
00107         self.ln   += 1
00108         # No sense in doing anything for an empty line or a comment line.
00109         if len(s) == 0 or match('^#',line): return None, None
00110         # From the line, figure out the interaction type.  If this line doesn't correspond
00111         # to an interaction, then do nothing.
00112         if s[0].upper() in pdict:
00113             self.itype = s[0].upper()
00114         # This is kind of a silly hack that allows us to take care of the 'multipole' keyword,
00115         # because of the syntax that the TINKER .prm file uses.
00116         elif s[0].upper() == 'MULTIPOLE':
00117             self.itype = 'MCHARGE'
00118         elif self.itype == 'MCHARGE':
00119             self.itype = 'DIPOLE'
00120         elif self.itype == 'DIPOLE':
00121             self.itype = 'QUADX'
00122         elif self.itype == 'QUADX':
00123             self.itype = 'QUADY'
00124         elif self.itype == 'QUADY':
00125             self.itype = 'QUADZ'
00126         else:
00127             self.itype = None
00128 
00129         if self.itype in pdict:
00130             if 'Atom' in pdict[self.itype]:
00131                 # List the atoms in the interaction.
00132                 self.atom = [s[i] for i in pdict[self.itype]['Atom']]
00133             # The suffix of the parameter ID is built from the atom    #
00134             # types/classes involved in the interaction.
00135             self.suffix = '.'.join(self.atom)
