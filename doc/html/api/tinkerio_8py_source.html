66  """

75  def feed(self, line):
76  """ Given a line, determine the interaction type and the atoms involved (the suffix).
77 
78  TINKER generally has stuff like this:
79 
80  @verbatim
81  bond-cubic -2.55
82  bond-quartic 3.793125
83 
84  vdw 1 3.4050 0.1100
85  vdw 2 2.6550 0.0135 0.910 # PARM 4
86 
87  multipole 2 1 2 0.25983
88  -0.03859 0.00000 -0.05818
89  -0.03673
90  0.00000 -0.10739
91  -0.00203 0.00000 0.14412
92  @endverbatim
93 
94  The '#PARM 4' has no effect on TINKER but it indicates that we
95  are tuning the fourth field on the line (the 0.910 value).
96 
97  @todo Put the rescaling factors for TINKER parameters in here.
98  Currently we're using the initial value to determine the
99  rescaling factor which is not very good.
100 
101  Every parameter line is prefaced by the interaction type
102  except for 'multipole' which is on multiple lines. Because
103  the lines that come after 'multipole' are predictable, we just
104  determine the current line using the previous line.
105 
106  Random note: Unit of force is kcal / mole / angstrom squared.
107 
108  """
109  s = line.split()
110  self.ln += 1
111  # No sense in doing anything for an empty line or a comment line.
112  if len(s) == 0 or match('^#',line): return None, None
113  # From the line, figure out the interaction type. If this line doesn't correspond
114  # to an interaction, then do nothing.
115  if s[0].upper() in pdict:
116  self.itype = s[0].upper()
117  # This is kind of a silly hack that allows us to take care of the 'multipole' keyword,
118  # because of the syntax that the TINKER .prm file uses.
119  elif s[0].upper() == 'MULTIPOLE':
120  self.itype = 'MCHARGE'
121  elif self.itype == 'MCHARGE':
122  self.itype = 'DIPOLE'
123  elif self.itype == 'DIPOLE':
124  self.itype = 'QUADX'
125  elif self.itype == 'QUADX':
126  self.itype = 'QUADY'
127  elif self.itype == 'QUADY':
128  self.itype = 'QUADZ'
129  else:
130  self.itype = None
131 
132  if self.itype in pdict:
133  if 'Atom' in pdict[self.itype]:
134  # List the atoms in the interaction.
135  self.atom = [s[i] for i in pdict[self.itype]['Atom']]
136  # The suffix of the parameter ID is built from the atom #
137  # types/classes involved in the interaction.
138  self.suffix = '.'.join(self.atom)

179 def modify_key(src, in_dict):
180  """ Performs in-place modification of a TINKER .key file.
181 
182  The input dictionary contains key:value pairs such as
183  "polarization direct". If the key exists in the TINKER file, then
184  that line is modified such that it contains the value in the
185  dictionary. Note that this "key" is not to be confused with the
186  .key extension in the TINKER file that we're modifying.
187

189  the dictionary should contain a None value or a blank space.
190 
191  If the key doesn't exist in the TINKER file, then the key:value pair
192  will be printed at the end.

194  @param[in] src Name of the TINKER file to be modified.
195  @param[in] in_dict Dictionary containing key-value pairs used to modify the TINKER file.
196 
197  """
198

200  fin = open(src).readlines()
201  else:
202  raise Exception("This function shouldn't be used to follow symbolic links, because I don't want to modify files in the target directory")
203  odict = OrderedDict([(key.lower(), val) for key, val in in_dict.items()])
204  flags = OrderedDict([(key, False) for key in odict.keys()])
205  outlines = []
206  for line in open(src).readlines():
207  s = line.split()
208  if len(s) == 0:
209  outlines.append(line)
210  continue
211  key = s[0].lower()
212  # Modify the line in-place if the key already exists.
213  if key in odict:
214  val = odict[key]
215  if val != None:
216  outlines.append("%s %s\n" % (key, val))
217  else:
218  outlines.append("%s\n" % (key))
219  flags[key] = True
220  else:
221  outlines.append(line)
222  for key, val in odict.items():
223  if not flags[key]:
224  if val != None:
225  outlines.append("%s %s\n" % (key, val))
226  else:
227  outlines.append("%s\n" % (key))
