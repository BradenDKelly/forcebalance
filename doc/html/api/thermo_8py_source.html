250  def objective_term(self, quantity):
251  """Calculates the contribution to the objective function (the term) for a
252  given quantity.
253 
254  Parameters
255  ----------
256  quantity : string
257  Calculate the objective term for this quantity.
258 
259  Returns
260  -------
261  term : dict
262  `term` is a dict with keys `X`, `G`, `H` and `info`. The values of
263  these keys are the objective term itself (`X`), its gradient (`G`),
264  its Hessian (`H`), and an OrderedDict with print information on
265  individiual data points (`info`).
266 
267  """
268  Objective = 0.0
269  Gradient = np.zeros(self.FF.np)
270  Hessian = np.zeros((self.FF.np, self.FF.np))
271 
272  # Grab ref data for quantity
273  qid = self.quantities.index(quantity)
274  Exp = np.array([pt.ref["refs"][qid] for pt in self.points])
275  Weights = np.array([pt.ref["weights"][qid] for pt in self.points])
276  Denom = self.denoms[quantity]
277 
278  # Renormalize weights
279  Weights /= np.sum(Weights)
280  logger.info("Renormalized weights to " + str(np.sum(Weights)) + "\n")
281  logger.info(("Physical quantity '%s' uses denominator = %g %s\n" %
282  (quantity.capitalize(), Denom,
283  self.points[0].ref["units"][self.quantities.index(quantity)])))
284 
285  # Grab calculated values
286  values = np.array([pt.data["values"][qid] for pt in self.points])
287  errors = np.array([pt.data["errors"][qid] for pt in self.points])
288  grads = np.array([pt.data["grads"][qid] for pt in self.points])
289 
290  # Calculate objective term using Least-squares function. Evaluate using
291  # Einstein summation: W is N-array, Delta is N-array and grads is
292  # NxM-array, where N is number of points and M is number of parameters.
293  #
294  # X_i = W_i * Delta2_i (no summed indices)
295  # G_ij = W_i * Delta_i * grads_ij (no summed indices)
296  # H_ijm = W_i * gradsT_jk * grads_lm (sum over k and l)
297  #
298  # Result: X is N-array, G is NxM-array and H is NxMxM-array.
299  #
300  Deltas = values - Exp
301  Objs = np.einsum("i,i->i", Weights, Deltas**2) / Denom / Denom
302  Grads = 2.0*np.einsum("i,i,ij->ij", Weights, Deltas, grads) / Denom / Denom
303  Hess = 2.0*np.einsum("i,jk,lm->ijm", Weights, grads.T, grads) / Denom / Denom
304 
305  # Average over all points
306  Objective += np.sum(Objs, axis=0)
307  Gradient += np.sum(Grads, axis=0)
308  Hessian += np.sum(Hess, axis=0)
309 
310  # Store gradients and setup print map
311  GradMapPrint = [["#Point"] + self.FF.plist]
312 
313  for pt in self.points:
314  temp = pt.temperature
315  press = pt.pressure
316  GradMapPrint.append([' %8.2f %8.1f' % (temp, press)] +
317  ["% 9.3e" % i for i in grads[pt.idnr-1]])
318 
319  o = wopen('gradient_%s.dat' % quantity)
320  for line in GradMapPrint:
321  print >> o, ' '.join(line)
322  o.close()
323 
324  printer = OrderedDict([(" %-5d %-12.2f %-8.1f"
325  % (pt.idnr, pt.temperature, pt.pressure),
326  ("% -10.3f % -10.3f +- %-8.3f % -8.3f % -9.5f % -9.5f"
327  % (Exp[pt.idnr-1], values[pt.idnr-1],
328  errors[pt.idnr-1], Deltas[pt.idnr-1],
329  Weights[pt.idnr-1], Objs[pt.idnr-1])))
330  for pt in self.points])
331

334  def get(self, mvals, AGrad=True, AHess=True):
335  """Return the contribution to the total objective function. This is a
336  weighted average of the calculated quantities.
337 
338  Parameters
339  ----------
340  mvals : list
341  Mathematical parameter values.
342  AGrad : Boolean
343  Switch to turn on analytic gradient.
344  AHess : Boolean
345  Switch to turn on analytic Hessian.
346 
347  Returns
348  -------
349  Answer : dict
350  Contribution to the objective function. `Answer` is a dict with keys
351  `X` for the objective function, `G` for its gradient and `H` for its
352  Hessian.
353 
354  """
355  Answer = {}
356 
357  Objective = 0.0
358  Gradient = np.zeros(self.FF.np)
359  Hessian = np.zeros((self.FF.np, self.FF.np))
360 
361  for pt in self.points:
362  # Update data point with MD results
363  self.retrieve(pt)
364 
365  obj = OrderedDict()
366  reweighted = []
367  for q in self.quantities:
368  # Returns dict with keys "X"=objective term value, "G"=the
369  # gradient, "H"=the hessian, and "info"=printed info about points
370  obj[q] = self.objective_term(q)
371 
372  # Apply weights for quantities (normalized)
373  if obj[q]["X"] == 0:
374  self.weights[q] = 0.0
375 
376  # Store weights sorted in the order of self.quantities
377  reweighted.append(self.weights[q])
378 
379  # Normalize weights
380  reweighted = np.array(reweighted)
381  wtot = np.sum(reweighted)
382  reweighted = reweighted/wtot if wtot > 0 else reweighted
383 
384  # Picks out the "X", "G" and "H" keys for the quantities sorted in the
385  # order of self.quantities. Xs is N-array, Gs is NxM-array and Hs is
386  # NxMxM-array, where N is number of quantities and M is number of
387  # parameters.
388  Xs = np.array([dic["X"] for dic in obj.values()])
389  Gs = np.array([dic["G"] for dic in obj.values()])
390  Hs = np.array([dic["H"] for dic in obj.values()])
391 
392  # Target contribution is (normalized) weighted averages of the
393  # individual quantity terms.
394  Objective = np.average(Xs, weights=(None if np.all(reweighted == 0) else \
395  reweighted), axis=0)
396  if AGrad:
397  Gradient = np.average(Gs, weights=(None if np.all(reweighted == 0) else \
398  reweighted), axis=0)
399  if AHess:
400  Hessian = np.average(Hs, weights=(None if np.all(reweighted == 0) else \
401  reweighted), axis=0)
402 
403  if not in_fd():
404  # Store results to show with indicator() function
405  self.Xp = {q : dic["X"] for (q, dic) in obj.items()}
406  self.Wp = {q : reweighted[self.quantities.index(q)]
407  for (q, dic) in obj.items()}
408  self.Pp = {q : dic["info"] for (q, dic) in obj.items()}
409 
410  if AGrad:
411  self.Gp = {q : dic["G"] for (q, dic) in obj.items()}
412 
413  self.Objective = Objective
414 
415  Answer = { "X": Objective, "G": Gradient, "H": Hessian }
416  return Answer
