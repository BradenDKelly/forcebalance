18 class FittingSimulation(ForceBalanceBaseClass):
19 
20  """
21  Base class for all fitting simulations.
22 
23  In ForceBalance a 'fitting simulation' is defined as a simulation
24  which computes a quantity that we can compare to a reference. The
25  force field parameters are tuned to reproduce the reference value as
26  closely as possible.
27 
28  The 'computable quantities' may include energies and forces where the
29  reference values come from QM calculations (energy and force matching),
30  energies from an EDA analysis (Maybe in the future, FDA?), molecular
31  properties (like polarizability, refractive indices, multipole moments
32  or vibrational frequencies), relative entropies, and bulk properties.
33  Single-molecule or bulk properties can even come from the experiment!
34 
35  The central idea in ForceBalance is that each quantity makes a
36  contribution to the overall objective function. So we can build force
37  fields that fit several quantities at once, rather than putting all of
38  our chips behind energy and force matching. In the future
39  ForceBalance may even include multiobjective optimization into the
40  optimizer.
41 
42  The optimization is done by way of minimizing an 'objective
43  function', which is comprised of squared differences between the
44  computed and reference values. These differences are not computed
45  in this file, but rather in subclasses that use FittingSimulation
46  as a base class. Thus, the contents of FittingSimulation itself
47  are meant to be as general as possible, because the pertinent
48  variables apply to all types of fitting simulations.
49 
50  An important node: FittingSimulation requires that all subclasses
51  have a method get(self,mvals,AGrad=False,AHess=False)
52  that does the following:
53 
54  Inputs:
55  mvals = The parameter vector, which modifies the force field
56  (Note to self: We include mvals with each FitSim because we can create
57  copies of the force field and do finite difference derivatives)
58  AGrad, AHess = Boolean switches for computing analytic gradients and Hessians
59 
60  Outputs:
61  Answer = {'X': Number, 'G': numpy.array(np), 'H': numpy.array((np,np)) }
62  'X' = The objective function itself
63  'G' = The gradient, elements not computed analytically are zero
64  'H' = The Hessian, elements not computed analytically are zero
65 
66  This is the only global requirement of a FittingSimulation.
67  Obviously 'get' itself is not defined here, because its
68  calculation will depend entirely on specifically which simulation
69  we wish to run. However, this should give us a unified framework
70  which will faciliate rapid implementation of FittingSimulations.
71 
72  Future work:
73  Robert suggested that I could enable automatic detection of which
74  parameters need to be computed by finite difference. Not a bad idea. :)
75 
76  """

165  def get_G(self,mvals=None):
166  """Computes the objective function contribution and its gradient.
167 
168  First the low-level 'get' method is called with the analytic gradient
169  switch turned on. Then we loop through the fd1_pids and compute
170  the corresponding elements of the gradient by finite difference,
171  if the 'fdgrad' switch is turned on. Alternately we can compute
172  the gradient elements and diagonal Hessian elements at the same time
173  using central difference if 'fdhessdiag' is turned on.
174  """
175  Ans = self.sget(mvals,1,0)
176  for i in range(self.FF.np):
177  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
178  if self.fdhessdiag:
179  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
180  elif self.fdgrad:
181  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
182  self.gct += 1
183  return Ans

185  def get_H(self,mvals=None):
186  """Computes the objective function contribution and its gradient / Hessian.
187 
188  First the low-level 'get' method is called with the analytic gradient
189  and Hessian both turned on. Then we loop through the fd1_pids and compute
190  the corresponding elements of the gradient by finite difference,
191  if the 'fdgrad' switch is turned on.
192 
193  This is followed by looping through the fd2_pids and computing the corresponding
194  Hessian elements by finite difference. Forward finite difference is used
195  throughout for the sake of speed.
196  """
197  Ans = self.sget(mvals,1,1)
198  if self.fdhess:
199  for i in range(self.FF.np):
200  if any([j in self.FF.plist[i] for j in self.fd1_pids]) or 'ALL' in self.fd1_pids:
201  Ans['G'][i] = f1d2p(fdwrap_G(self,mvals,i),self.h,f0 = Ans['X'])
202  for i in range(self.FF.np):
203  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
204  FDSlice = f1d2p(fdwrap_H(self,mvals,i),self.h,f0 = Ans['G'])
205  Ans['H'][i,:] = FDSlice
206  Ans['H'][:,i] = FDSlice
207  elif self.fdhessdiag:
208  for i in range(self.FF.np):
209  if any([j in self.FF.plist[i] for j in self.fd2_pids]) or 'ALL' in self.fd2_pids:
210  Ans['G'][i], Ans['H'][i,i] = f12d3p(fdwrap_G(self,mvals,i),self.h, f0 = Ans['X'])
211  self.hct += 1
212  return Ans
