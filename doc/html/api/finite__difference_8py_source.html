8 def f1d2p(f, h, f0 = None):
9  """
10  A two-point finite difference stencil.
11  This function does either two computations or one,
12  depending on whether the 'center' value is supplied.
13  This is done in order to avoid recomputing the center
14  value many times when we repeat this function for each
15  index of the gradient.
16 
17  How to use: use fdwrap or something similar to generate
18  a one-variable function from the (usually) much more complicated
19  function that we wish to differentate. Then pass it to this function.
20 
21  Inputs:
22  f = The one-variable function f(x) that we're differentiating
23  h = The finite difference step size, usually a small number
24 
25  Outputs:
26  fp = The finite difference derivative of the function f(x) around x=0.
27  """
28  if f0 == None:
29  f0, f1 = [f(i*h) for i in [0, 1]]
30  else:
31  f1 = f(h)
32  fp = (f1-f0)/h
33  return fp

35 def f1d5p(f, h):
36  """
37  A highly accurate five-point finite difference stencil
38  for computing derivatives of a function. It works on both
39  scalar and vector functions (i.e. functions that return arrays).
40  Since the function does four computations, it's costly but
41  recommended if we really need an accurate reference value.
42 
43  The function is evaluated at points -2h, -h, +h and +2h
44  and these values are combined to make the derivative according to:
45  http://www.holoborodko.com/pavel/numerical-methods/numerical-derivative/central-differences/
46 
47  How to use: use fdwrap or something similar to generate
48  a one-variable function from the (usually) much more complicated
49  function that we wish to differentate. Then pass it to this function.
50 
51  Inputs:
52  f = The one-variable function f(x) that we're differentiating
53  h = The finite difference step size, usually a small number
54 
55  Outputs:
56  fp = The finite difference derivative of the function f(x) around x=0.
57  """
58  fm2, fm1, f1, f2 = [f(i*h) for i in [-2, -1, 1, 2]]
59  fp = (-1*f2+8*f1-8*fm1+1*fm2)/(12*h)
60  return fp

77 def f12d3p(f, h, f0 = None):
78  """
79  A three-point finite difference stencil.
80  This function does either two computations or three,
81  depending on whether the 'center' value is supplied.
82  This is done in order to avoid recomputing the center
83  value many times.
84 
85  The first derivative is evaluated using central difference.
86  One advantage of using central difference (as opposed to forward
87  difference) is that we get zero at the bottom of a parabola.
88 
89  Using this formula we also get an approximate second derivative, which
90  can then be inserted into the diagonal of the Hessian. This is very
91  useful for optimizations like BFGS where the diagonal determines
92  how far we step in the parameter space.
93 
94  How to use: use fdwrap or something similar to generate
95  a one-variable function from the (usually) much more complicated
96  function that we wish to differentate. Then pass it to this function.
97 
98  Inputs:
99  f = The one-variable function f(x) that we're differentiating
100  h = The finite difference step size, usually a small number
101 
102  Outputs:
103  fp = The finite difference derivative of the function f(x) around x=0.
104  """
105  if f0 == None:
106  fm1, f0, f1 = [f(i*h) for i in [-1, 0, 1]]
107  else:
108  fm1, f1 = [f(i*h) for i in [-1, 1]]
109  fp = (f1-fm1)/(2*h)
110  fpp = (fm1-2*f0+f1)/(h*h)
111  return fp, fpp

113 def f2var(f, h):
114  """ A finite difference stencil for a function of two variables. """
115  fpp, fpm, fmp, fmm = [f(i*h, j*h) for i, j in [(1,1), (1,-1), (-1,1), (-1,-1)]]
116  return (fpp-fpm-fmp+fmm)/(4*h**2)

130 def fdwrap(func,mvals0,pidx,key=None,**kwargs):
131  """
132  A function wrapper for finite difference designed for
133  differentiating 'get'-type functions.
134 
135  Since our finite difference stencils take single-variable functions
136  and differentiate them around zero, and our objective function is
137  quite a complicated function, we need a wrapper to serve as a
138  middleman. The alternative would be to copy the finite difference
139  formula to wherever we're taking the derivative, and that is prone
140  to mistakes.
141 
142  Inputs:
143  func = Either get_X or get_G; these functions return dictionaries. ['X'] = 1.23, ['G'] = [0.12, 3,45, ...]
144  mvals0 = The 'central' values of the mathematical parameters - i.e. the wrapped function's origin is here.
145  pidx = The index of the parameter that we're differentiating
146  key = either 'G' or 'X', the value we wish to take out of the dictionary
147  kwargs = Anything else we want to pass to the objective function (for instance, Project.Objective takes Order as an argument)
148 
149  Outputs:
150  func1 = Wrapped version of func, which takes a single float argument.
151  """
152  def func1(arg):
153  mvals = list(mvals0)
154  mvals[pidx] += arg
155  logger.info("\rfdwrap: " + func.__name__ + " [%i] = % .1e " % (pidx, arg) + ' '*50 + '\r')
156  if key != None:
157  return func(mvals,**kwargs)[key]
158  else:
159  return func(mvals,**kwargs)
160  return func1
