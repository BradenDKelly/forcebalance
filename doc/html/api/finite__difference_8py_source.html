00035 def f1d5p(f, h):
00036     """
00037     A highly accurate five-point finite difference stencil
00038     for computing derivatives of a function.  It works on both
00039     scalar and vector functions (i.e. functions that return arrays).
00040     Since the function does four computations, it's costly but
00041     recommended if we really need an accurate reference value.
00042 
00043     The function is evaluated at points -2h, -h, +h and +2h
00044     and these values are combined to make the derivative according to:
00045     http://www.holoborodko.com/pavel/numerical-methods/numerical-derivative/central-differences/
00046 
00047     How to use: use fdwrap or something similar to generate
00048     a one-variable function from the (usually) much more complicated
00049     function that we wish to differentate.  Then pass it to this function.
00050 
00051     Inputs:
00052     f  = The one-variable function f(x) that we're differentiating
00053     h  = The finite difference step size, usually a small number
00054 
00055     Outputs:
00056     fp = The finite difference derivative of the function f(x) around x=0.
00057     """
00058     fm2, fm1, f1, f2 = [f(i*h) for i in [-2, -1, 1, 2]]

00077 def f12d3p(f, h, f0 = None):
00078     """
00079     A three-point finite difference stencil.
00080     This function does either two computations or three,
00081     depending on whether the 'center' value is supplied.
00082     This is done in order to avoid recomputing the center
00083     value many times.
00084 
00085     The first derivative is evaluated using central difference.
00086     One advantage of using central difference (as opposed to forward
00087     difference) is that we get zero at the bottom of a parabola.
00088 
00089     Using this formula we also get an approximate second derivative, which
00090     can then be inserted into the diagonal of the Hessian.  This is very
00091     useful for optimizations like BFGS where the diagonal determines
00092     how far we step in the parameter space.
00093 
00094     How to use: use fdwrap or something similar to generate
00095     a one-variable function from the (usually) much more complicated
00096     function that we wish to differentate.  Then pass it to this function.
00097 
00098     Inputs:
00099     f  = The one-variable function f(x) that we're differentiating
00100     h  = The finite difference step size, usually a small number
00101 
00102     Outputs:
00103     fp = The finite difference derivative of the function f(x) around x=0.
00104     """
00105     if f0 == None:
00106         fm1, f0, f1 = [f(i*h) for i in [-1, 0, 1]]
00107     else:
00108         fm1, f1 = [f(i*h) for i in [-1, 1]]
00109     fp = (f1-fm1)/(2*h)
00110     fpp = (fm1-2*f0+f1)/(h*h)
00111     return fp, fpp

00119 def fdwrap(func,mvals0,pidx,key=None,**kwargs):
00120     """
00121     A function wrapper for finite difference designed for
00122     differentiating 'get'-type functions.
00123 
00124     Since our finite difference stencils take single-variable functions
00125     and differentiate them around zero, and our objective function is
00126     quite a complicated function, we need a wrapper to serve as a
00127     middleman.  The alternative would be to copy the finite difference
00128     formula to wherever we're taking the derivative, and that is prone
00129     to mistakes.
00130 
00131     Inputs:
00132     func   = Either get_X or get_G; these functions return dictionaries. ['X'] = 1.23, ['G'] = [0.12, 3,45, ...]
00133     mvals0 = The 'central' values of the mathematical parameters - i.e. the wrapped function's origin is here.
00134     pidx   = The index of the parameter that we're differentiating
00135     key    = either 'G' or 'X', the value we wish to take out of the dictionary
00136     kwargs = Anything else we want to pass to the objective function (for instance, Project.Objective takes Order as an argument)
00137 
00138     Outputs:
00139     func1  = Wrapped version of func, which takes a single float argument.
00140     """
00141     def func1(arg):
00142         mvals = list(mvals0)
00143         mvals[pidx] += arg
00144         logger.info("\rfdwrap: " + func.__name__ + " [%i] = % .1e " % (pidx, arg) + ' '*50)
00145         if key != None:
00146             return func(mvals,**kwargs)[key]
00147         else:
00148             return func(mvals,**kwargs)
00149     return func1
