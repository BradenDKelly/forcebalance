00063     ##
00064     #  Given a line, determine the interaction type and the atoms involved (the suffix).
00065     # 
00066     #         TINKER generally has stuff like this:
00067     # 
00068     #         @verbatim
00069     #         bond-cubic              -2.55
00070     #         bond-quartic            3.793125
00071     # 
00072     #         vdw           1               3.4050     0.1100
00073     #         vdw           2               2.6550     0.0135      0.910 # PARM 4
00074     # 
00075     #         multipole     2    1    2               0.25983
00076     #                                                -0.03859    0.00000   -0.05818
00077     #                                                -0.03673
00078     #                                                 0.00000   -0.10739
00079     #                                                -0.00203    0.00000    0.14412
00080     #         @endverbatim
00081     # 
00082     #         The '#PARM 4' has no effect on TINKER but it indicates that we
00083     #         are tuning the fourth field on the line (the 0.910 value).
00084     # 
00085     #         @todo Put the rescaling factors for TINKER parameters in here.
00086     #         Currently we're using the initial value to determine the
00087     #         rescaling factor which is not very good.
00088     # 
00089     #         Every parameter line is prefaced by the interaction type
00090     #         except for 'multipole' which is on multiple lines.  Because
00091     #         the lines that come after 'multipole' are predictable, we just
00092     #         determine the current line using the previous line.
00093     # 
00094     #         Random note: Unit of force is kcal / mole / angstrom squared.
00095     #         
00096     #         
00097     def feed(self, line):
00098         s          = line.split()
00099         self.ln   += 1
00100         # No sense in doing anything for an empty line or a comment line.
00101         if len(s) == 0 or match('^#',line): return None, None
00102         # From the line, figure out the interaction type.  If this line doesn't correspond
00103         # to an interaction, then do nothing.
00104         if s[0].upper() in pdict:
00105             self.itype = s[0].upper()
00106         # This is kind of a silly hack that allows us to take care of the 'multipole' keyword,
00107         # because of the syntax that the TINKER .prm file uses.
00108         elif s[0].upper() == 'MULTIPOLE':
00109             self.itype = 'CHARGE'
00110         elif self.itype == 'CHARGE':
00111             self.itype = 'DIPOLE'
00112         elif self.itype == 'DIPOLE':
00113             self.itype = 'QUADX'
00114         elif self.itype == 'QUADX':
00115             self.itype = 'QUADY'
00116         elif self.itype == 'QUADY':
00117             self.itype = 'QUADZ'
00118         else:
00119             self.itype = None
00120 
00121         if self.itype in pdict:
00122             if 'Atom' in pdict[self.itype]:
00123                 # List the atoms in the interaction.
00124                 self.atom = [s[i] for i in pdict[self.itype]['Atom']]
00125             # The suffix of the parameter ID is built from the atom    #
00126             # types/classes involved in the interaction.
00127             self.suffix = '.'.join(self.atom)
