00002 #  @package parser Input file parser for ForceBalance projects.  Additionally, the location for all default options.
00003 # 
00004 # Although I will do my best to write good documentation,
00005 # for many programs the input parser becomes the most up-to-date
00006 # source for documentation.  So this is a great place to write
00007 # lots of comments for those who implement new functionality.
00008 # 
00009 # Basically, the way my program is structured there is a set of
00010 # GENERAL options and also a set of SIMULATION options.  Since
00011 # there can be many fitting simulations within a single project
00012 # (i.e. we may wish to fit water trimers and hexamers, which
00013 # constitutes two fitting simulations) the input is organized
00014 # into sections, like so:
00015 # 
00016 # $options\n
00017 # gen_option_1 Big\n
00018 # gen_option_2 Mao\n
00019 # $simulation\n
00020 # sim_option_1 Sniffy\n
00021 # sim_option_2 Schmao\n
00022 # $simulation\n
00023 # sim_option_1 Nifty\n
00024 # sim_option_2 Jiffy\n
00025 # $end
00026 # 
00027 # In this case, two sets of simulation options are generated in addition to the general option.
00028 # 
00029 # Each option is meant to be parsed as a certain variable type.
00030 # 
00031 # - String option values are read in directly; note that only the first two words in the line are processed
00032 # - Some strings are capitalized when they are read in; this is mainly for function tables like OptTab and SimTab
00033 # - List option types will pick up all of the words on the line and use them as values,
00034 # plus if the option occurs more than once it will aggregate all of the values.
00035 # - Integer and float option types are read in a pretty straightforward way
00036 # - Boolean option types are always set to true, unless the second word is '0', 'no', or 'false' (not case sensitive)
00037 # - Section option types are meant to treat more elaborate inputs, such
00038 # as the user pasting in output parameters from a previous job as input,
00039 # or a specification of internal coordinate system.  I imagine that for
00040 # every section type I would have to write my own parser.  Maybe a
00041 # ParsTab of parsing functions would work. :)
00042 # 
00043 # To add a new option, simply add it to the dictionaries below and give it a default value if desired.
00044 # If you add an entirely new type, make sure to implement the interpretation of that type in the parse_inputs function.
00045 # 
00046 # @author Lee-Ping Wang
00047 # @date 12/2011
00048 #

00166 ##
00167 #  Print out a section of the input file in a parser-compliant and readable format.
00168 # 
00169 #     At the time of writing of this function, it's mainly intended to be called by MakeInputFile.py.
00170 #     The heading is printed first (it is something like $options or $simulation).  Then it loops
00171 #     through the variable types (strings, allcaps, etc...) and the keys in each variable type.
00172 #     The one-line description of each key is printed out as a comment, and then the key itself is
00173 #     printed out along with the value provided in optdict.  If optdict is None, then the default
00174 #     value is printed out instead.
00175 # 
00176 #     @param[in] heading Heading, either $options or $simulation
00177 #     @param[in] optdict Options dictionary or None.
00178 #     @param[in] typedict Option type dictionary, either gen_opts_types or sim_opts_types specified in this file.
00179 #     @return Answer List of strings for the section that we are printing out.
00180 #     
00181 #     
00182 def printsection(heading,optdict,typedict):
00183     Answer = [heading]
00184     firstentry = 1
00185     for i in ['strings','allcaps','lists','ints','bools','floats','sections']:
00186         vartype = re.sub('s$','',i)
00187         for j in typedict[i]:
00188             val = optdict[j] if optdict != None else typedict[i][j][0]
00189             if firstentry:
00190                 firstentry = 0
00191             else:
00192                 Answer.append("")
00193             Answer.append("# (%s) %s" % (vartype, typedict[i][j][1]))
00194             Answer.append("%s %s" % (str(j),str(val)))
00195     Answer.append("$end")
00196     return Answer

00198 ##
00199 #  Parse through the input file and read all user-supplied options.
00200 # 
00201 #     This is usually the first thing that happens when an executable script is called.
00202 #     Our parser first loads the default options, and then updates these options as it
00203 #     encounters keywords.
00204 # 
00205 #     Each keyword corresponds to a variable type; each variable type (e.g. string,
00206 #     integer, float, boolean) is treated differently.  For more elaborate inputs,
00207 #     there is a 'section' variable type.
00208 # 
00209 #     There is only one set of general options, but multiple sets of fitting simulation options.
00210 #     Each fitting simulation has its own section delimited by the \em $simulation keyword,
00211 #     and we build a list of simulation options.  
00212 # 
00213 #     @param[in]  input_file The name of the input file.
00214 #     @return     options    General options.
00215 #     @return     sim_opts   List of fitting simulation options.
00216 #     
00217 #     @todo Implement internal coordinates.
00218 #     @todo Implement sampling correction.
00219 #     @todo Implement charge groups.
00220 #     
00221 def parse_inputs(input_file):
00222     
00223     print "Reading options from file: %s" % input_file
00224     section = "NONE"
00225     # First load in all of the default options.
00226     options = {'root':os.getcwd()}
00227     options.update(gen_opts_defaults)
00228     sim_opts = []
00229     this_sim_opt = deepcopy(sim_opts_defaults)
00230     fobj = open(input_file)
00231     for line in fobj:
00232         # Anything after "#" is a comment
00233         line = line.split("#")[0].strip()
00234         s = line.split()
00235         # Skip over blank lines
00236         if len(s) == 0:
00237             continue
00238         key = s[0].lower()
00239         # If line starts with a $, this signifies that we're in a new section.
00240         if re.match('^\$',line):
00241             newsection = re.sub('^\$','',line).upper()
00242             if section == "SIMULATION" and newsection in mainsections:
00243                 sim_opts.append(this_sim_opt)
00244                 this_sim_opt = deepcopy(sim_opts_defaults)
00245             section = newsection
00246         elif section in ["OPTIONS","SIMULATION"]:
00247             ## Depending on which section we are in, we choose the correct type dictionary
00248             ## and add stuff to 'options' and 'this_sim_opt'
00249             (this_opt, opts_types) = (options, gen_opts_types) if section == "OPTIONS" else (this_sim_opt, sim_opts_types)
00250             ## Note that "None" is a special keyword!  The variable will ACTUALLY be set to None.
00251             if len(s) > 1 and s[1].upper() == "NONE":
00252                 this_opt[key] = None
00253             elif key in opts_types['strings']:
00254                 this_opt[key] = s[1]
00255             elif key in opts_types['allcaps']:
00256                 this_opt[key] = s[1].upper()
00257             elif key in opts_types['lists']:
00258                 for word in s[1:]:
00259                     this_opt.setdefault(key,[]).append(word)
00260             elif key in opts_types['ints']:
00261                 this_opt[key] = int(s[1])
00262             elif key in opts_types['bools']:
00263                 if len(s) == 1:
00264                     this_opt[key] = True
00265                 elif s[1].upper() in ["0", "NO", "FALSE"]:
00266                     this_opt[key] = False
00267                 else:
00268                     this_opt[key] = True
00269             elif key in opts_types['floats']:
00270                 this_opt[key] = float(s[1])
00271             elif key in opts_types['sections']:
00272                 this_opt[key] = ParsTab[key](fobj)
00273             else:
00274                 print "Unrecognized keyword: --- \x1b[1;91m%s\x1b[0m --- in %s section" \
00275                       % (key, section == "OPTIONS" and "general" or "simulation")
00276                 print "Perhaps this option actually belongs in the %s section?" \
00277                       % (section == "OPTIONS" and "simulation" or "general")
00278                 sys.exit(1)
00279         elif section not in mainsections:
00280             print "Unrecognized section: %s" % section
00281             sys.exit(1)
00282     if section == "SIMULATION":
00283         sim_opts.append(this_sim_opt)
00284     return options, sim_opts
