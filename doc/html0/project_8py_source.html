00015 # Container for a ForceBalance force field optimization project.
00016 # 
00017 #     The triumvirate or trinity of components are:
00018 #     - The force field
00019 #     - The objective function
00020 #     - The optimizer
00021 # 
00022 #     The force field is a class defined in forcefield.py.
00023 #     The objective function is built here as a combination of fitting simulation classes.
00024 #     The optimizer is a class defined in this file.
00025 #     
00026 class Project(object):
00027     
00028     ##
00029     # Instantiation of a ForceBalance force field optimization project.
00030     # 
00031     #         Here's what we do:
00032     # 
00033     #         - Parse the input file
00034     #         - Create an instance of the force field
00035     #         - Create a list of fitting simulation instances
00036     #         - Create an optimizer instance
00037     #         - Print out the general options
00038     #         
00039     #         
00040     def __init__(self,input_file):
00041         #======================================#
00042         #    Call these functions at startup   #
00043         #======================================#
00044         ## The general options and simulation options that come from parsing the input file
00045         self.options, self.sim_opts = parse_inputs(input_file)
00046         ## The force field component of the project
00047         self.FF = FF(self.options)
00048         ## The list of fitting simulations
00049         self.Simulations = [SimTab[opts['simtype']](self.options,opts,self.FF) for opts in self.sim_opts]
00050         ## The optimizer component of the project
00051         self.Optimizer   = Optimizer(self.options,self.Objective,self.FF,self.Simulations)
00052         
00053     ##
00054     #  Objective function defined within Project; can you think of a better place?
00055     # 
00056     #         The objective function is a combination of contributions from the different
00057     #         fitting simulations.  Basically, it loops through the fitting simulations,
00058     #         gets their contributions to the objective function and then sums all of them
00059     #         (although more elaborate schemes are conceivable).  The return value is the
00060     #         same data type as calling the fitting simulation itself: a dictionary containing
00061     #         the objective function, the gradient and the Hessian.
00062     # 
00063     #         The penalty function is also computed here; it keeps the parameters from straying
00064     #         too far from their initial values.
00065     # 
00066     #         @param[in] mvals The mathematical parameters that enter into computing the objective function
00067     #         @param[in] Order The requested order of differentiation
00068     #         @param[in] usepvals Switch that determines whether to use physical parameter values
00069     #         
00070     def Objective(self,mvals,Order=0,usepvals=False,verbose=False):
00071         ## This is the objective function; it's a dictionary containing the value, first and second derivatives
00072         Objective = {'X':0.0, 'G':zeros(self.FF.np), 'H':zeros((self.FF.np,self.FF.np))}
00073         ## This is the canonical lettering that corresponds to : objective function, gradient, Hessian.
00074         Letters = ['X','G','H']
00075         ## Obtain the denominator.
00076         WTot = sum([i.weight for i in self.Simulations])
00077         # Loop through the simulations.
00078         for Sim in self.Simulations:
00079             # The first call is always done at the midpoint.
00080             Sim.bSave = True
00081             # List of functions that I can call.
00082             Funcs   = [Sim.get_X, Sim.get_G, Sim.get_H]
00083             # Call the appropriate function
00084             Ans = Funcs[Order](mvals)
00085             # Print out the qualitative indicators
00086             if verbose:
00087                 Sim.indicate()
00088             # Note that no matter which order of function we call, we still increment the objective / gradient / Hessian the same way.
00089             for i in range(3):
00090                 Objective[Letters[i]] += Ans[Letters[i]]*Sim.weight/WTot
00091         ## Compute the penalty function.
00092         Extra = Penalty(mvals,Objective,self.options['penalty_type'],self.options['penalty_additive'],self.options['penalty_multiplicative'])
00093         for i in range(3):
00094             Objective[Letters[i]] += Extra[i]
00095         return Objective
