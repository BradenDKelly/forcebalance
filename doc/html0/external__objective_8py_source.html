00005 # This is a module that computes the objective function from an external program
00006 # which does not have ForTune-functionality built in.  Logically, all we have to do
00007 # is get finite-difference derivatives of the energy and force for a single snapshot,
00008 # and then build up the objective function by looping over snapshots.  Hence, we need
00009 # an objective-function builder and a finite-difference wrapper.
00010 # 
00011 # It is designed to be (hopefully) as general as possible, so that the only modules
00012 # that need to interface to this program consist of:
00013 # 
00014 # 1) A subroutine to automatically write input files for (program)
00015 # 2) A subroutine that calls (program)
00016 # 3) A subroutine to parse output files for (program)

00087 def singular_matrix_inverse(M):
00088     tiny = 1e-3
00089     u,s,vh = linalg.svd(M)
00090     for i in range(s.shape[0]):
00091         if abs(s[i]) > 1e-3:
00092             s[i] = 1/s[i]

00097 ##
00098 # 
00099 #     LPW 04-23-2011
00100 #     This subroutine is designed to serve the same function as "get" in foptutils9.py,
00101 #     except it does not depend on an external program to build the objective function
00102 #     and its derivatives.  Most of my force matching is done in my ForTune-GROMACS build,
00103 #     which computes the objective function and prints it out; then "get" only needs to
00104 #     call GROMACS and read it in.
00105 #     
00106 #     Here, the external program is only expected to give the energy, forces, and their
00107 #     derivatives.  Now this subroutine may sound trivial since the objective function is
00108 #     simply a least-squares quantity (M-Q)^2 - but there are a number of nontrivial
00109 #     considerations.  I will list them here.
00110 #     0) Polytensor formulation: Because there may exist covariance between different
00111 #     components of the force (or covariance between the energy and the force), we build
00112 #     the objective function by taking outer products of vectors that have the form
00113 #     [E F_1x F_1y F_1z F_2x F_2y ... ], and then we trace it with the inverse of the
00114 #     covariance matrix to get the objective function.
00115 #     1) Boltzmann weights and normalization: Each snapshot has its own Boltzmann weight,
00116 #     which may or may not be normalized.  This subroutine does the normalization automatically.
00117 #     2) Subtracting out the mean energy gap: The zero-point energy difference between
00118 #     reference and fitting simulations is meaningless.  This subroutine subtracts it out.
00119 #     3) Hybrid ensembles: This program builds a combined objective function from both
00120 #     MM and QM ensembles, which is rigorously better than using a single ensemble.
00121 #     
00122 #     Note that this subroutine does not do EVERYTHING that ForTune-GROMACS can do, which
00123 #     include:
00124 #     1) Internal coordinate systems
00125 #     2) 'Sampling correction' (deprecated, since it doesn't seem to improve upon much.. heh)
00126 #     
00127 #     Since this subroutine is designed to take advantage of a semi-empirical code such
00128 #     as MOPAC, which contains no derivatives, it is designed to work with another
00129 #     subroutine that calls MOPAC with a set of parameters/coordinates and returns
00130 #     the energies/forces and their derivatives.  This other subroutine should be a finite
00131 #     difference wrapper to MOPAC.
00132 def get_external(parms, pkg, First=1, Second=0):
00133     # Timing stuff
00134     T0 = time.time()
00135     cwd = os.getcwd()
00136     os.chdir(pkg.root + '/temp')
00137 
00138     NC   = 3*pkg.fitatoms
00139     NCP1 = 3*pkg.fitatoms+1
00140     NP   = pkg.np
00141     # Note: Because of hybrid ensembles, we form two separate objective functions.
00142     # This means the code is (trivially) doubled in length because there are two sets
00143     # of Boltzmann weights.  In principle I could write a general subroutine for a single
00144     # set of Boltzmann weights and call it twice, but that would require me to loop over
00145     # the snapshots twice.  However, after the loop over snapshots, the subroutine that
00146     # builds the objective function (build_objective, above) is called twice using the
00147     # MM-ensemble and QM-ensemble variables.
00148 
00149     # Partition functions.
00150     Z       = 0
00151     Y       = 0
00152 
00153     Q = zeros(NCP1,dtype=float)
00154     
00155     ### STEP 1: Form all of the arrays. ###
00156     # Quantities pertaining to the covariance matrix.
00157     # An explanation of variable names follows.
00158     # M0_M  = The mean of the MM values in the MM ensemble
00159     # M0_Q_p= The mean of the MM first derivatives in the QM ensemble
00160     # QQ_Q  = The unnormalized expected value of <Q(X)Q>
00161     # Later on we divide these quantities by Z to normalize,
00162     # and Q0(X)Q0 is subtracted from QQ to get the covariance.
00163     M0_M    = zeros(NCP1,dtype=float)
00164     Q0_M    = zeros(NCP1,dtype=float)
00165     QQ_M    = zeros((NCP1,NCP1),dtype=float)
00166     M0_M_p  = zeros((NP,NCP1),dtype=float)
00167     M0_M_pq = zeros((NP,NP,NCP1),dtype=float)
00168     
00169     M0_Q    = zeros(NCP1,dtype=float)
00170     Q0_Q    = zeros(NCP1,dtype=float)
00171     QQ_Q    = zeros((NCP1,NCP1),dtype=float)
00172     M0_Q_p  = zeros((NP,NCP1),dtype=float)
00173     M0_Q_pq = zeros((NP,NP,NCP1),dtype=float)
00174 
00175     # Objective function polytensors:
00176     # This is formed in a loop over snapshots by taking the outer product
00177     # (Q-M)(X)(Q-M), multiplying by the Boltzmann weight, and then summing.
00178     
00179     SPiXi = zeros((NCP1,NCP1),dtype=float)
00180     SPiXi_p  = zeros((NP,NCP1,NCP1),dtype=float)
00181     SPiXi_pq = zeros((NP,NP,NCP1,NCP1),dtype=float)
00182     
00183     SRiXi = zeros((NCP1,NCP1),dtype=float)
00184     SRiXi_p  = zeros((NP,NCP1,NCP1),dtype=float)
00185     SRiXi_pq = zeros((NP,NP,NCP1,NCP1),dtype=float)
00186 
00187     QBN = dot(pkg.qmboltz[:pkg.shots],pkg.whamboltz[:pkg.shots])
00188 
00189     # If we are doing parallel optimization, farm out the jobs
00190     splsize = pkg.paropt_splitsize
00191     if pkg.paropt:
00192         from fortune import SGE
00193         
00194         # serialize the information needed by the slaves that is not already on disk
00195         parenvin = open('parenvin', 'w')
00196         pickle.dump({'pkg':pkg, 'parms':parms, 'first':First, 'second':Second, 'splitsize':splsize}, parenvin)
00197         parenvin.close()
00198 
00199         njobs = int(ceil(pkg.shots/double(splsize)))
00200         
00201         cluster = SGE()
00202         jobid = cluster.submit('runext.py $SGE_TASK_ID', stdout='ext_out.$SGE_TASK_ID', stderr='ext_err.$SGE_TASK_ID', name='ext_opt', nproc = 1, arraysize = njobs, verbose = False)     
00203         cluster.wait(jobid, interval = 0.2, verbose=False)
00204 
00205 
00206     ### STEP 2: Loop through the snapshots. ###
00207     for i in range(pkg.shots):
00208         # Build Boltzmann weights and increment partition function.
00209         P   = pkg.whamboltz[i]
00210         Z  += P
00211         R   = pkg.qmboltz[i]*pkg.whamboltz[i] / QBN
00212         Y  += R
00213         #print "R: % .8f" % R
00214         # Recall reference (QM) data
00215         Q[0] = pkg.eqm[i]
00216         Q[1:] = pkg.fqmfit[i*NC:(i+1)*NC]
00217         # Increment the average quantities.
00218         QQ     = outer(Q,Q)
00219         
00220         # if we are not doing parallel optimization, do the normal call
00221         if not pkg.paropt:
00222             M, M_p, M_pq = call_external_program(pkg,parms,i,program=pkg.external_program,First=First,Second=Second)
00223         # otherwise, load the derivatives from the hard drive
00224         # this could be faster, since each of these files is quite large (can approach megabytes)
00225         # ideally, we would communicate with sockets or some other such fanciness
00226         else:
00227             arrs = numpy.load('ext_%d.npz' % i)
00228             M = arrs['M']
00229             if First:
00230                 M_p = arrs['M_p']
00231             if Second:
00232                 M_pq = arrs['M_pq']
00233 
00234         M0_M += P*M
00235         Q0_M += P*Q
00236         QQ_M += P*QQ
00237         M0_Q += R*M
00238         Q0_Q += R*Q
00239         QQ_Q += R*QQ
00240         # Increment the objective function.
00241         MmQ = M - Q
00242         #Xi  = outer(MmQ,MmQ)
00243         Xi  = outer(M,M) - 2*outer(Q,M) + outer(Q,Q)
00244         SPiXi += P * Xi
00245         SRiXi += R * Xi
00246         if not First: continue
00247         for p in range(NP):
00248             # Increment the first derivatives of the average quantities.
00249             M0_M_p[p] += P * M_p[p]
00250             M0_Q_p[p] += R * M_p[p]
00251             # Increment the first derivatives of the objective function polytensor.
00252             Xi_p        = 2 * outer(MmQ,M_p[p])
00253             SPiXi_p[p] += P * Xi_p
00254             SRiXi_p[p] += R * Xi_p
00255             for q in range(NP):
00256                 # Increment the second derivatives of the objective function polytensor.
00257                 # NOTE: Even if we take first derivatives of the energy and force, there is an important contribution to the second derivatives of the objective function!
00258                 Xi_pq         = 2 * (outer(M_p[p],M_p[q]))
00259                 if Second:
00260                     Xi_pq    += 2 * (outer(MmQ,M_pq[p,q]))
00261                 SPiXi_pq[p,q] += P * Xi_pq
00262                 SRiXi_pq[p,q] += R * Xi_pq
00263                 if Second:
00264                     # Increment the second derivatives of the average quantities.
00265                     M0_M_pq[p,q] += P * M_pq[p,q]
00266                     M0_Q_pq[p,q] += R * M_pq[p,q]
00267 
00268     ### STEP 3: Build the covariance matrix and invert it. ###
00269     # The "two" is of no importance, it has been kept historically.
00270     EFW     = 2 *pkg.efweight
00271     CEFW    = 2 - EFW
00272     # Build the weight matrix, so the force contribution is suppressed by 1/3N
00273     WM      = zeros((NCP1,NCP1),dtype=float)
00274     WM[0,0] = sqrt(EFW)
00275     for i in range(1,NCP1):
00276         WM[i,i] = sqrt(EFW/NC)
00277     # The covariance matrix
00278     # Note: There is a detail here that might appear peculiar at first:
00279     # I'm building only one covariance matrix from both ensembles.
00280     # This is the proper thing to do, believe me.
00281     # The functional form looks like (X2_M + X2_Q)(C)^-1
00282     # I will build X2_M C^-1 and X2_Q C^-1 separately because it's cleaner that way.
00283     QBP  = pkg.qmboltzporp
00284     MBP  = 1 - pkg.qmboltzporp
00285     C    = MBP*(QQ_M-outer(Q0_M,Q0_M)/Z)/Z + QBP*(QQ_Q-outer(Q0_Q,Q0_Q)/Y)/Y
00286     # Get rid of energy-force covariance
00287     Ci   = singular_matrix_inverse(C)
00288     for i in range(1,NCP1):
00289         Ci[0,i] = 0.0;
00290         Ci[i,0] = 0.0;
00291     WCiW = array(matrix(WM) * matrix(Ci) * matrix(WM)) # Weighted covariance matrix.
00292     
00293     ### STEP 4: Build the objective function and its derivatives by tracing with the covariance matrix. ###
00294     # (Note that without Sampling Correction, covariance matrix has no derivatives.  I am thankful for this.)
00295     # I will build X2_M C^-1 and X2_Q C^-1 separately because it's cleaner that way.
00296     X2_M, X2_M_p, X2_M_pq  = build_objective(SPiXi,SPiXi_p,SPiXi_pq,WCiW,Z,Q0_M,M0_M,M0_M_p,M0_M_pq,NP,NCP1,First=First,Second=Second)
00297     X2_Q, X2_Q_p, X2_Q_pq  = build_objective(SRiXi,SRiXi_p,SRiXi_pq,WCiW,Y,Q0_Q,M0_Q,M0_Q_p,M0_Q_pq,NP,NCP1,First=First,Second=Second)
00298

00300     ## for i in range(NCP1):
00301     ##     for j in range(NCP1):
00302     ##         print "% 8.6e" % SPiXi[i,j],
00303     ##     print
00304     ## print
00305     ## for j in range(NCP1):
00306     ##     print "% 12.6f" % M0_M[j],
00307     ## print
00308     ## for j in range(NCP1):
00309     ##     print "% 12.6f" % Q0_M[j],
00310     ## print
00311     ## print "Objective Function MM: % 12.6e" % X2_M,
00312     ## print "Analytic Second Derivative, MM:"
00313     ## print
00314     ## for i in range(NP):
00315     ##     for j in range(NP):
00316     ##         print "% 8.6e" % X2_M_pq[i,j],
00317     ##     print
00318     ## exit()

00320     ### STEP 5: These are the quantities that get passed out of GROMACS. ###
00321     # BC   = The un-penalized objective function!
00322     # BC1R = The first derivative of the objective function
00323     # BC2R = The second derivative of the objective function
00324     BC   = MBP * X2_M    + QBP * X2_Q
00325     BC1R = MBP * X2_M_p  + QBP * X2_Q_p
00326     BC2R = MBP * X2_M_pq + QBP * X2_Q_pq
00327     # Energy error in kJ/mol
00328     E0_M = (2*Q0_M[0]*M0_M[0] - Q0_M[0]*Q0_M[0] - M0_M[0]*M0_M[0])/Z/Z;
00329     E0_Q = (2*Q0_Q[0]*M0_Q[0] - Q0_Q[0]*Q0_Q[0] - M0_Q[0]*M0_Q[0])/Y/Y;
00330     E     = MBP * sqrt(SPiXi[0,0]/Z + E0_M) + QBP * sqrt(SRiXi[0,0]/Y + E0_Q)
00331     # Fractional energy error.  This is not used right now but I thought it might help to have the formula.
00332     # Note: SPiXi and QQ both need to be divided by Z to be truly "averaged", but since we're dividing them, the Z cancels out.
00333     Efrac = MBP * sqrt(SPiXi[0,0]/QQ_M[0,0]) + QBP * sqrt(SRiXi[0,0]/QQ_Q[0,0])
00334     # Fractional force error; this is the internal force error,
00335     # and for us this only means it is Boltzmann-averaged,
00336     # because we are not doing the sampling correction or the internal coordinates.
00337     # This is given by the root-mean-square diagonal objective function values (divided by the diagonal covariance values).
00338     F     = MBP * sqrt(mean(array([SPiXi[i,i]/QQ_M[i,i] for i in range(1,NCP1)]))) + QBP * sqrt(mean(array([SRiXi[i,i]/QQ_Q[i,i] for i in range(1,NCP1)])))

00341     ## print BC
00342     ## print BC1R
00343     ## print BC2R
