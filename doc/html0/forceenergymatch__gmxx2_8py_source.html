00027 # ForceBalance class for force and energy matching with the modified GROMACS.
00028 # 
00029 #     This class allows us to use a heavily modified version of GROMACS
00030 #     (a major component of this program) to compute the objective
00031 #     function contribution.  The modified GROMACS does the looping
00032 #     through snapshots, computes the interactions as well as the
00033 #     derivatives, and sums them up to build the objective function.
00034 #     I will write that documentation elsewhere, perhaps when I port
00035 #     GROMACS over to version 4.5.4.
00036 # 
00037 #     This class implements the 'get' method.  When 'get' is called, the
00038 #     force field is printed to the temporary directory along with
00039 #     several files containing the information needed by the modified
00040 #     GROMACS (the Boltzmann weights, the parameters that need
00041 #     derivatives and their values, the QM energies and forces, and the
00042 #     energy / force weighting.)
00043 # 
00044 #     The modified GROMACS is called with the arguments '-rerun all.gro
00045 #     -fortune -rerunvsite' to loop over the snapshots, turn on force
00046 #     matching functionality and reconstruct virtual site positions (an
00047 #     important consideration if we're changing the virtual site
00048 #     positions in the optimization).  Its outputs are 'e2f2bc' which
00049 #     means 'energy squared, force squared, boltzmann corrected' and
00050 #     contains the objective function, 'a1dbc' and 'a2dbc' containing
00051 #     analytic first and second derivatives, 'gmxboltz' containing the
00052 #     Boltzmann weights used, and possibly some other stuff.
00053 # 
00054 #     Most importantly, 'e2f2bc', 'a1dbc' and 'a2dbc' are read by 'get'
00055 #     after GROMACS is called and returned directly as the objective
00056 #     function contributions.
00057 # 
00058 #     Other methods implemented in this class are related to the
00059 #     preparation of the temp directory.
00060 #     
00061 #

00174     # 
00175     #         First we create the force field using the parameter values that were
00176     #         passed in.  Note that we may pass in physical parameters directly
00177     #         and bypass the coordinate transformation by setting self.usepvals to True.
00178     # 
00179     #         The physical parameters are printed to 'pvals' for GROMACS to read -
00180     #         of course GROMACS knows the parameters already, but this facilitates
00181     #         retrieval from the low level subroutines.
00182     # 
00183     #         Several switches are printed to files, such as:
00184     #         - 'FirstDerivativesOnly' to prevent computation of the Hessian
00185     #         - 'NoDerivatives' to prevent computation of the Hessian AND the gradient
00186     # 
00187     #         GROMACS is called in the callgmxx2() method.
00188     # 
00189     #         The output files are then parsed for the objective function and its
00190     #         derivatives are read in.  The answer is passed out as a dictionary:
00191     #         {'X': Objective Function, 'G': Gradient, 'H': Hessian}
00192     # 
00193     #         @param[in] mvals Mathematical parameter values
00194     #         @param[in] AGrad Switch to turn on analytic gradient
00195     #         @param[in] AHess Switch to turn on analytic Hessian
00196     #         @param[in] tempdir Temporary directory for running computation
00197     #         @return Answer Contribution to the objective function
00198     # 
00199     #         @todo Some of these files don't need to be printed, they can be passed
00200     #         to GROMACS as arguments.  Let's think about this some more.
00201     #         @todo Currently I have no way to pass out the qualitative indicators.

00204     def get(self,mvals,AGrad=False,AHess=False,tempdir=None):
00205         if tempdir == None:
00206             tempdir = self.tempdir
00207         Answer = {}
00208         cwd = os.getcwd()
00209         # Create the new force field!!
00210         pvals = self.FF.make(tempdir,mvals,self.usepvals)
00211         gmxx2_print(os.path.join(os.path.join(self.root,tempdir,"pvals")),append([0],pvals),"double")
00212         os.chdir(os.path.join(self.root,tempdir))
00213         if AHess:
00214             AGrad = True
00215             remove_if_exists("FirstDerivativesOnly")
00216             remove_if_exists("NoDerivatives")
00217         elif AGrad:
00218             with open("FirstDerivativesOnly",'w') as f: f.close()
00219             remove_if_exists("NoDerivatives")
00220         else:
00221             with open("NoDerivatives",'w') as f: f.close()
00222         print "GMXX2: %s\r" % tempdir,
00223         self.callgmxx2()
00224         # Parse the output files
00225         for line in open('e2f2bc').readlines():
00226             sline = line.split()
00227             if match('^Objective Function',line):
00228                 X = floatornan(sline[-1])
00229                 Answer['X'] = X
00230                 # Pass out the qualitative indicator
00231             elif match('^Energy Error \(kJ/mol\)',line):
00232                 E = floatornan(sline[-1])
00233                 self.e_err = E
00234             elif match('^Force Error',line):
00235                 F = floatornan(sline[-1])
00236                 self.f_err = F
00237         # Derivatives in the physical parameter values
00238         GR = array([float(i.split()[2]) for i in open('a1dbc').readlines()])
00239         # If we're working in the rescaled space, rescale and transform to the 'mathematical' derivatives
00240         # Otherwise just use the raw values.
00241         if self.usepvals:
00242             Answer['G'] = col(GR.copy())
00243         else:
00244             G = flat(mat(self.FF.tm)*col(GR.copy()))
00245             Answer['G'] = G
00246         # Reads in second derivatives from GROMACS
00247         # Even if we're just calling the gradients, we want the diagonal Hessian elements as well!
00248         HR = zeros((self.FF.np, self.FF.np))
00249         for line in open('a2dbc').readlines():
00250             i = int(line.split()[1])
00251             j = int(line.split()[2])
00252             HR[i, j] = float(line.split()[3])
00253         if self.usepvals:
00254             Answer['H'] = mat(HR.copy())
00255         else:
00256             H = mat(self.FF.tm)*mat(HR.copy())*mat(self.FF.tmI)                       # Projects charge constraints out of second derivative
00257             Answer['H'] = H
00258         os.chdir(cwd)
00259         return Answer
