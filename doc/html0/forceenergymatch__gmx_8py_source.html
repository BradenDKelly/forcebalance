00042 # ForceBalance class for force and energy matching with the modified GROMACS.
00043 # 
00044 #     This class allows us to use a heavily modified version of GROMACS
00045 #     (a major component of this program) to compute the objective
00046 #     function contribution.  The modified GROMACS does the looping
00047 #     through snapshots, computes the interactions as well as the
00048 #     derivatives, and sums them up to build the objective function.
00049 #     I will write that documentation elsewhere, perhaps when I port
00050 #     GROMACS over to version 4.5.4.
00051 # 
00052 #     This class implements the 'get' method.  When 'get' is called, the
00053 #     force field is printed to the temporary directory along with
00054 #     several files containing the information needed by the modified
00055 #     GROMACS (the Boltzmann weights, the parameters that need
00056 #     derivatives and their values, the QM energies and forces, and the
00057 #     energy / force weighting.)
00058 # 
00059 #     The modified GROMACS is called with the arguments '-rerun all.gro
00060 #     -fortune -rerunvsite' to loop over the snapshots, turn on force
00061 #     matching functionality and reconstruct virtual site positions (an
00062 #     important consideration if we're changing the virtual site
00063 #     positions in the optimization).  Its outputs are 'e2f2bc' which
00064 #     means 'energy squared, force squared, boltzmann corrected' and
00065 #     contains the objective function, 'a1dbc' and 'a2dbc' containing
00066 #     analytic first and second derivatives, 'gmxboltz' containing the
00067 #     Boltzmann weights used, and possibly some other stuff.
00068 # 
00069 #     Most importantly, 'e2f2bc', 'a1dbc' and 'a2dbc' are read by 'get'
00070 #     after GROMACS is called and returned directly as the objective
00071 #     function contributions.
00072 # 
00073 #     Other methods implemented in this class are related to the
00074 #     preparation of the temp directory.
00075 #     
00076 #

00259     #  Calls the modified GROMACS and collects the objective function contribution.
00260     # 
00261     #         First we create the force field using the parameter values that were
00262     #         passed in.  Note that we may pass in physical parameters directly
00263     #         and bypass the coordinate transformation by setting self.usepvals to True.
00264     # 
00265     #         The physical parameters are printed to 'pvals' for GROMACS to read -
00266     #         of course GROMACS knows the parameters already, but this facilitates
00267     #         retrieval from the low level subroutines.
00268     # 
00269     #         Several switches are printed to files, such as:
00270     #         - 'FirstDerivativesOnly' to prevent computation of the Hessian
00271     #         - 'NoDerivatives' to prevent computation of the Hessian AND the gradient
00272     # 
00273     #         GROMACS is called in the callgmx() method.
00274     # 
00275     #         The output files are then parsed for the objective function and its
00276     #         derivatives are read in.  The answer is passed out as a dictionary:
00277     #         {'X': Objective Function, 'G': Gradient, 'H': Hessian}
00278     # 
00279     #         @param[in] mvals Mathematical parameter values
00280     #         @param[in] AGrad Switch to turn on analytic gradient

00282     #         @param[in] tempdir Temporary directory for running computation
00283     #         @return Answer Contribution to the objective function
00284     # 
00285     #         @todo Some of these files don't need to be printed, they can be passed
00286     #         to GROMACS as arguments.  Let's think about this some more.
00287     #         @todo Currently I have no way to pass out the qualitative indicators.
00288     #

00290     def get(self,mvals,AGrad=False,AHess=False,tempdir=None):
00291         if tempdir == None:
00292             tempdir = self.tempdir
00293         Answer = {}
00294         cwd = os.getcwd()
00295         # Create the new force field!!
00296         pvals = self.FF.make(tempdir,mvals,self.usepvals)
00297         gmxprint(os.path.join(os.path.join(self.root,tempdir,"pvals")),append([0],pvals),"double")
00298         os.chdir(os.path.join(self.root,tempdir))
00299         if AHess:
00300             AGrad = True
00301             remove_if_exists("FirstDerivativesOnly")
00302             remove_if_exists("NoDerivatives")
00303         elif AGrad:
00304             with open("FirstDerivativesOnly",'w') as f: f.close()
00305             remove_if_exists("NoDerivatives")
00306         else:
00307             with open("NoDerivatives",'w') as f: f.close()
00308         print "GMX: %s\r" % tempdir,
00309         self.callgmx()
00310         # Parse the output files
00311         for line in open('e2f2bc').readlines():
00312             sline = line.split()
00313             if match('^Objective Function',line):
00314                 X = floatornan(sline[-1])
00315                 Answer['X'] = X
00316                 # Pass out the qualitative indicator
00317             elif match('^Energy Error \(kJ/mol\)',line):
00318                 E = floatornan(sline[-1])
00319                 self.e_err = E
00320             elif match('^Force Error',line):
00321                 F = floatornan(sline[-1])
00322                 self.f_err = F
00323         # Derivatives in the physical parameter values
00324         GR = array([float(i.split()[2]) for i in open('a1dbc').readlines()])
00325         # If we're working in the rescaled space, rescale and transform to the 'mathematical' derivatives
00326         # Otherwise just use the raw values.
00327         if self.usepvals:
00328             Answer['G'] = col(GR.copy())
00329         else:
00330             G = flat(mat(self.FF.tm)*col(GR.copy()))
00331             Answer['G'] = G
00332         # Reads in second derivatives from GROMACS
00333         # Even if we're just calling the gradients, we want the diagonal Hessian elements as well!
00334         HR = zeros((self.FF.np, self.FF.np))
00335         for line in open('a2dbc').readlines():
00336             i = int(line.split()[1])
00337             j = int(line.split()[2])
00338             HR[i, j] = float(line.split()[3])
00339         if self.usepvals:
00340             Answer['H'] = mat(HR.copy())
00341         else:
00342             H = mat(self.FF.tm)*mat(HR.copy())*mat(self.FF.tmI)                       # Projects charge constraints out of second derivative
00343             Answer['H'] = H
00344         os.chdir(cwd)
00345         return Answer
