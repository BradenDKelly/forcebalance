00091 #  Parses the 'atomtype' line.
00092 #     
00093 #     Parses lines like this:\n
00094 #     <tt> opls_135     CT    6   12.0107    0.0000    A    3.5000e-01    2.7614e-01\n
00095 #     C       12.0107    0.0000    A    3.7500e-01    4.3932e-01\n
00096 #     Na  11    22.9897    0.0000    A    6.068128070229e+03  2.662662556402e+01  0.0000e+00 ; PARM 5 6\n </tt>
00097 #     Look at all the variety!
00098 # 
00099 #     @param[in] line Input line.
00100 #     @return answer Dictionary containing:\n
00101 #     atom type\n
00102 #     bonded atom type (if any)\n
00103 #     atomic number (if any)\n
00104 #     atomic mass\n
00105 #     charge\n
00106 #     particle type\n
00107 #     force field parameters\n
00108 #     number of optional fields
00109 #     
00110 def parse_atomtype_line(line):
00111     # First split the line up to the comment.  We don't care about the comment at this time
00112     sline = line.split(';')[0].split()
00113     # The line must contain at least six fields to be considered data.
00114     if len(sline) < 6:
00115         return
00116     # Using variable "wrd" because the line has a variable number of fields
00117     # Can you think of a better way?
00118     wrd = 0
00119     bonus = 0
00120     atomtype = sline[wrd]
00121     batomtype = sline[wrd]
00122     wrd += 1
00123     # The bonded atom type, a pecularity of OPLS-AA
00124     # Test if it begins with a letter.  Seems to work. :)
00125     if match('[A-Za-z]',sline[wrd]):
00126         batomtype = sline[wrd]
00127         wrd += 1
00128         bonus += 1
00129     # Now to test if the next line is an atomic number or a mass.
00130     # Atomic numbers never have decimals...
00131     atomicnum = -1
00132     if isint(sline[wrd]):
00133         atomicnum = int(sline[wrd])
00134         wrd += 1
00135         bonus += 1
00136     # The mass can be overridden in the 'atoms' section.
00137     mass = float(sline[wrd])
00138     wrd += 1
00139     # Atom types have a default charge though this is almost always overridden
00140     chg  = float(sline[wrd])
00141     wrd += 1
00142     # Particle type. Actual atom or virtual site?
00143     ptp  = sline[wrd]
00144     wrd += 1
00145     param = [float(i) for i in sline[wrd:]]
00146     answer = {'atomtype':atomtype, 'batomtype':batomtype, 'atomicnum':atomicnum, 'mass':mass, 'chg':chg, 'ptp':ptp, 'param':param, 'bonus':bonus}
00147     return answer

00177     #  Given a line, determine the interaction type and the atoms involved (the suffix).
00178     #         
00179     #         For example, we want \n
00180     #         <tt> H    O    H    5    1.231258497536e+02    4.269161426840e+02   -1.033397697685e-02   1.304674117410e+04 ; PARM 4 5 6 7 </tt> \n
00181     #         to give us itype = 'UREY_BRADLEY' and suffix = 'HOH'
00182     #         
00183     #         If we are in a TypeSection, it returns a list of atom types; \n
00184     #         If we are in a TopolSection, it returns a list of atom names.
00185     #         
00186     #         The section is essentially a case statement that picks out the
00187     #         appropriate interaction type and makes a list of the atoms
00188     #         involved
00189     # 
00190     #         Note that we can call gmxdump for this as well, but I
00191     #         prefer to read the force field file directly.
00192     #         
00193     #         ToDo: [ atoms ] section might need to be more flexible to accommodate optional fields
00194     #         
00195     #         
00196     def feed(self, line):
00197         s          = line.split()
00198         atom       = []
00199         self.itype = None
00200         self.ln   += 1
00201         # No sense in doing anything for an empty line or a comment line.
00202         if len(s) == 0 or match('^;',line): return None, None
00203         # Now go through all the cases.
00204         if match('^\[.*\]',line):
00205             # Makes a word like "atoms", "bonds" etc.
00206             self.sec = sub('[\[\] \n]','',line)
00207         elif self.sec == 'defaults':
00208             self.nbtype = int(s[0])
00209         elif self.sec == 'moleculetype':
00210             self.res    = s[0]
00211         elif self.sec == 'atomtypes':
00212             atype = parse_atomtype_line(line)
00213             # Basically we're shifting the word positions
00214             # based on the syntax of the line in 'atomtype', but it allows the parameter typing to
00215             # keep up with the flexibility of the syntax of these lines.
00216             if atype['bonus'] > 0:
00217                 pdict['VDW'] = {4+atype['bonus']:'S',5+atype['bonus']:'T'}
00218                 pdict['VDW_BHAM'] = {4+atype['bonus']:'A', 5+atype['bonus']:'B', 6+atype['bonus']:'C'}
00219             atom = atype['atomtype']
00220             self.itype = fdict[self.sec][self.nbtype]
00221         elif self.sec == 'nonbond_params':
00222             atom = [s[0], s[1]]
00223             self.itype = pftypes[self.nbtype]
00224         elif self.sec == 'atoms':
00225             atom = [s[4]]
00226             self.itype = 'COUL'
00227             # Build the adict here.
00228             self.adict.setdefault(s[3],[]).append(s[4])
00229         elif self.sec == 'qtpie':
00230             # The atom involved is labeled by the atomic number.
00231             atom = [s[0]]
00232             self.itype = 'QTPIE'
00233         elif self.sec == 'bonds':
00234             atom = [self.adict[self.res][int(i) - 1] for i in s[:2]]
00235             self.itype = fdict[self.sec][int(s[2])]
00236         elif self.sec == 'bondtypes':
00237             atom = [s[0], s[1]]
00238             self.itype = fdict[self.sec][int(s[2])]
00239         elif self.sec == 'angles':
00240             atom = [self.adict[self.res][int(i) - 1] for i in s[:3]]
00241             self.itype = fdict[self.sec][int(s[3])]
00242         elif self.sec == 'angletypes':
00243             atom = [s[0], s[1], s[2]]
00244             self.itype = fdict[self.sec][int(s[3])]
00245         elif self.sec == 'dihedrals':
00246             atom = [self.adict[self.res][int(i)-1] for i in s[:4]]
00247             self.itype = fdict[self.sec][int(s[4])]
00248             if self.itype == 'PDIHS' and len(s) >= 7:
00249                 # Add the multiplicity of the dihedrals to the interaction type.
00250                 self.itype += s[7]
00251         elif self.sec == 'dihedraltypes':
00252             atom = [s[0], s[1], s[2], s[3]]
00253             self.itype = fdict[self.sec][int(s[4])]
00254             if self.itype == 'PDIHS' and len(s) >= 7:
00255                 self.itype += s[7]
00256         elif self.sec == 'virtual_sites2':
00257             atom = [s[0]]
00258             self.itype = fdict[self.sec][int(s[3])]
00259         elif self.sec == 'virtual_sites3':
00260             atom = [s[0]]
00261             self.itype = fdict[self.sec][int(s[4])]
00262         elif self.sec == 'virtual_sites4':
00263             atom = [s[0]]
00264             self.itype = fdict[self.sec][int(s[5])]
00265         else:
00266             return [],"Confused"
